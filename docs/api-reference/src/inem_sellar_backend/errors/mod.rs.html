26/// Error unificado de la API de InemSellar.
27///
28/// # Por que `#[derive(Error)]`
29/// `Error` viene del crate `thiserror`. Genera automaticamente la
30/// implementacion del trait `std::error::Error` para nuestro enum.
31/// Sin esto, tendriamos que implementar `Error`, `Display` y `Debug`
32/// a mano — unas 30 lineas de boilerplate por variante.
33///
34/// # Por que `#[error("...")]` en cada variante
35/// `thiserror` usa estos atributos para generar el `Display` del error.
36/// El texto entre comillas es lo que devuelve `self.to_string()`.
37/// `{0}` se reemplaza por el primer campo de la variante (interpolacion).
38///
39/// # Por que `#[from]` en `Database`
40/// `#[from]` genera automaticamente `impl From<sqlx::Error> for AppError`.
41/// Esto permite usar el operador `?` en handlers: si una query SQLx falla,
42/// el error se convierte automaticamente en `AppError::Database(...)`.
43/// Es como si Rust hiciera `.map_err(AppError::Database)` por ti.
44///
45/// # Equivalencia en Dart
46/// En Dart harias algo como:
47/// ```dart
48/// sealed class AppError {
49///   const factory AppError.notFound(String msg) = NotFoundError;
50///   const factory AppError.badRequest(String msg) = BadRequestError;
51///   // ...
52/// }
53/// ```
54/// En Rust, un `enum` con `thiserror` logra lo mismo pero con pattern matching
55/// exhaustivo: el compilador te obliga a manejar TODAS las variantes.
56#[derive(Error, Debug)]
57pub enum AppError {
58    /// Recurso no encontrado (HTTP 404).
59    /// Uso: `AppError::NotFound("Usuario con id X".into())`
60    #[error("Recurso no encontrado: {0}")]
61    NotFound(String),
62
63    /// Datos de entrada invalidos (HTTP 400).
64    /// Uso: `AppError::BadRequest("Email no valido".into())`
65    #[error("Datos invalidos: {0}")]
66    BadRequest(String),
67
68    /// No autenticado — falta token JWT o es invalido (HTTP 401).
69    /// No tiene campo String porque el mensaje siempre es el mismo.
70    #[error("No autenticado")]
71    Unauthorized,
72
73    /// Autenticado pero sin permisos para esta accion (HTTP 403).
74    /// Ej: un usuario normal intenta acceder a un endpoint de admin.
75    #[error("Sin permisos para esta accion")]
76    Forbidden,
77
78    /// Conflicto con el estado actual (HTTP 409).
79    /// Ej: intentar registrar un email que ya existe (violacion UNIQUE).
80    #[error("Conflicto: {0}")]
81    Conflict(String),
82
83    /// Error de base de datos (HTTP 500).
84    /// `#[from]` permite que `?` convierta sea_orm::DbErr en AppError.
85    #[error(transparent)]
86    Database(#[from] sea_orm::DbErr),
87
88    /// Error interno generico (HTTP 500).
89    #[error("Error interno: {0}")]
90    Internal(String),
91}
92
93impl AppError {
94    /// Metodo helper para convertir DbErr en AppError.
95    /// Util con `.map_err(AppError::from_db)` en los repositorios.
96    pub fn from_db(err: sea_orm::DbErr) -> Self {
97        AppError::Database(err)
98    }
99}

101/// Implementacion de `EndpointOutRegister` para que `#[endpoint]` pueda documentar
102/// las respuestas de error en la especificacion OpenAPI generada automaticamente.
103///
104/// # Por que este trait
105/// Cuando un handler marcado con `#[endpoint]` devuelve `Result<T, AppError>`,
106/// Salvo necesita saber que codigos HTTP puede retornar el error para incluirlos
107/// en la documentacion OpenAPI. Este trait le dice a Salvo: "este error puede
108/// producir 400, 401, 403, 404, 409 y 500".
109///
110/// # Por que `let _ = components`
111/// `components` se usa para registrar schemas de tipos complejos (e.g., structs
112/// de respuesta de error con cuerpo JSON documentado). En nuestro caso solo
113/// registramos descripciones textuales simples, asi que no necesitamos `components`.
114/// La asignacion `let _ = ...` suprime el warning del compilador de variable no usada.
115impl EndpointOutRegister for AppError {
116    fn register(components: &mut Components, operation: &mut Operation) {
117        // Usamos String::from("400") en vez de "400".into() para ayudar al compilador
118        // a resolver la ambiguedad de tipos cuando hay multiples impls de Into<String>.
119        operation.responses.insert(
120            String::from("400"),
121            salvo::oapi::Response::new(
122                "Datos invalidos — el cuerpo JSON o los parametros no son correctos",
123            ),
124        );
125        operation.responses.insert(
126            String::from("401"),
127            salvo::oapi::Response::new("No autenticado — falta el token JWT o es invalido"),
128        );
129        operation.responses.insert(
130            String::from("403"),
131            salvo::oapi::Response::new(
132                "Sin permisos — el usuario autenticado no tiene acceso a este recurso",
133            ),
134        );
135        operation.responses.insert(
136            String::from("404"),
137            salvo::oapi::Response::new("Recurso no encontrado"),
138        );
139        operation.responses.insert(
140            String::from("409"),
141            salvo::oapi::Response::new("Conflicto — el recurso ya existe (ej: email duplicado)"),
142        );
143        operation.responses.insert(
144            String::from("500"),
145            salvo::oapi::Response::new("Error interno del servidor"),
146        );
147        // No registramos schemas de error en components porque las respuestas
148        // de error son objetos JSON simples { error: string, codigo: number }
149        // que no requieren definicion formal de schema.
150        let _ = components;
151    }
152}

154/// # Por que implementar `Writer`
155/// `Writer` es el trait de Salvo que convierte un tipo en una respuesta HTTP.
156/// Cuando un handler devuelve `Result<Json<T>, AppError>`, Salvo llama a
157/// `AppError::write()` si hay error. Aqui decidimos:
158/// - Que status code HTTP devolver (404, 401, 500, etc.)
159/// - Que JSON mandar al cliente
160///
161/// # Seguridad: no exponer errores internos
162/// Para `Database` e `Internal`, logueamos el error real con `tracing::error!`
163/// (se ve en los logs del servidor) pero al cliente le mandamos un mensaje
164/// generico "Error interno del servidor". Esto evita filtrar nombres de tablas,
165/// queries SQL, o stack traces al frontend.
166///
167/// # Por que `#[async_trait]`
168/// `Writer::write` es una funcion async dentro de un trait. Rust aun usa
169/// el crate `async_trait` para esto en muchas librerias (incluido Salvo).
170/// `#[async_trait]` transforma la funcion async en un tipo que Rust puede
171/// manejar en un trait. Salvo re-exporta este macro desde su prelude.
172#[async_trait]
173impl Writer for AppError {
174    async fn write(self, _req: &mut Request, _depot: &mut Depot, res: &mut Response) {
175        // Determinamos status code y mensaje segun la variante.
176        // `match` es exhaustivo: si anadimos una nueva variante a AppError
177        // y olvidamos manejarla aqui, el compilador da error.
178        let (status, mensaje) = match &self {
179            AppError::NotFound(msg) => (StatusCode::NOT_FOUND, msg.clone()),
180            AppError::BadRequest(msg) => (StatusCode::BAD_REQUEST, msg.clone()),
181            AppError::Unauthorized => (StatusCode::UNAUTHORIZED, "No autenticado".to_string()),
182            AppError::Forbidden => (
183                StatusCode::FORBIDDEN,
184                "Sin permisos para esta accion".to_string(),
185            ),
186            AppError::Conflict(msg) => (StatusCode::CONFLICT, msg.clone()),
187
188            // SEGURIDAD: errores de BD e internos se loguean pero NO se exponen
189            AppError::Database(err) => {
190                tracing::error!("Error de base de datos: {:?}", err);
191                (
192                    StatusCode::INTERNAL_SERVER_ERROR,
193                    "Error interno del servidor".to_string(),
194                )
195            }
196            AppError::Internal(msg) => {
197                tracing::error!("Error interno: {}", msg);
198                (
199                    StatusCode::INTERNAL_SERVER_ERROR,
200                    "Error interno del servidor".to_string(),
201                )
202            }
203        };
204
205        // Escribimos la respuesta HTTP:
206        // 1. Ponemos el status code (404, 401, 500, etc.)
207        // 2. Renderizamos un JSON con el mensaje de error
208        //
209        // `Json(...)` es un wrapper de Salvo que serializa a JSON y pone
210        // el header Content-Type: application/json automaticamente.
211        //
212        // `serde_json::json!` es un macro que crea un serde_json::Value
213        // a partir de una sintaxis similar a JSON literal.
214        res.status_code(status);
215        res.render(Json(serde_json::json!({
216            "error": mensaje,
217            "codigo": status.as_u16(),
218        })));
219    }
220}
