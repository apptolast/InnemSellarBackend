5// QUE ES: define el arbol de rutas de la API REST.
6// Conecta URLs con handlers usando el Router de Salvo.
7//
8// # Patron de Salvo para rutas
9// Salvo usa un arbol de routers composable:
10// - `Router::with_path("x")` crea un nodo con path "x"
11// - `.get(handler)` asocia un handler al metodo GET
12// - `.push(Router::...)` anade un hijo (ruta anidada)
13// - `.hoop(middleware)` aplica middleware a ese nodo y sus hijos
14//
15// Es similar a go_router en Flutter:
16//   GoRoute(path: '/comunidades', builder: listaComunidades,
17//     routes: [GoRoute(path: ':id', builder: detalleComunidad)])
18
19use salvo::prelude::*;
20
21use crate::handlers::{auth, geografia, ofertas};
22use crate::middleware;
23
24/// Crea el router completo de la API.
25///
26/// Estructura de URLs:
27/// ```text
28/// /                                         → hello (health check)
29/// /api/v1/auth/registro                     → POST registro
30/// /api/v1/auth/login                        → POST login
31/// /api/v1/auth/refrescar                    → POST refrescar token
32/// /api/v1/auth/logout                       → POST logout (auth requerida)
33/// /api/v1/comunidades                       → GET listar
34/// /api/v1/comunidades/{id}                  → GET obtener por id
35/// /api/v1/provincias                        → GET listar (?id_comunidad=X)
36/// /api/v1/provincias/{id}                   → GET obtener por id
37/// /api/v1/provincias/{id}/oficina           → GET oficina SEPE
38/// /api/v1/ofertas                           → GET listar (publico)
39/// /api/v1/ofertas/{id}                      → GET obtener (publico)
40/// /api/v1/ofertas                           → POST crear (auth)
41/// /api/v1/ofertas/{id}                      → PUT actualizar (auth, solo autor)
42/// /api/v1/ofertas/{id}                      → DELETE eliminar (auth, solo autor)
43/// ```
44///
45/// # Por que `{id}` y no `:id` o `<id>`
46/// Salvo v0.76+ usa la sintaxis `{nombre}` para path params.
47/// `:id` es de Express/Actix, `<id>` era la sintaxis antigua de Salvo.
48/// `{id:num}` restringiria a numeros, pero no lo usamos porque
49/// nuestro param es i32 y la conversion falla con 400 si no es numero.
50pub fn crear_router() -> Router {
51    Router::new().push(
52        Router::with_path("api/v1")
53            // /api/v1/auth — rutas publicas (sin middleware auth)
54            .push(
55                Router::with_path("auth")
56                    .push(Router::with_path("registro").post(auth::registro))
57                    .push(Router::with_path("login").post(auth::login))
58                    .push(Router::with_path("refrescar").post(auth::refrescar))
59                    .push(
60                        // logout requiere auth
61                        Router::with_path("logout")
62                            .hoop(middleware::auth_middleware)
63                            .post(auth::logout),
64                    ),
65            )
66            // /api/v1/comunidades — rutas publicas
67            .push(
68                Router::with_path("comunidades")
69                    .get(geografia::listar_comunidades)
70                    .push(
71                        // /api/v1/comunidades/{id}
72                        Router::with_path("{id}").get(geografia::obtener_comunidad),
73                    ),
74            )
75            // /api/v1/provincias
76            .push(
77                Router::with_path("provincias")
78                    .get(geografia::listar_provincias)
79                    .push(
80                        Router::with_path("{id}")
81                            .get(geografia::obtener_provincia)
82                            .push(
83                                Router::with_path("oficina")
84                                    .get(geografia::obtener_oficina_por_provincia),
85                            ),
86                    ),
87            )
88            // /api/v1/ofertas — GET publico, POST/PUT/DELETE protegido
89            .push(
90                Router::with_path("ofertas")
91                    .get(ofertas::listar_ofertas)
92                    .push(
93                        // POST crear — requiere auth
94                        Router::new()
95                            .hoop(middleware::auth_middleware)
96                            .post(ofertas::crear_oferta),
97                    )
98                    .push(
99                        Router::with_path("{id}").get(ofertas::obtener_oferta).push(
100                            // PUT/DELETE — requieren auth
101                            Router::new()
102                                .hoop(middleware::auth_middleware)
103                                .put(ofertas::actualizar_oferta)
104                                .delete(ofertas::eliminar_oferta),
105                        ),
106                    ),
107            ),
108    )
109}
