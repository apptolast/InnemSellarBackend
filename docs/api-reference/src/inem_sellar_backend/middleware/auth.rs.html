1// src/middleware/auth.rs
2//
3// Middleware de autenticacion JWT para Salvo.
4//
5// # Como funciona un middleware en Salvo
6// Un middleware es un handler que decide si la peticion puede continuar.
7// Si el token es valido: inserta el id_usuario en Depot y llama a call_next.
8// Si no: responde 401 y llama a skip_rest (no llega al handler).
9//
10// Es como un middleware de Express.js o un interceptor de Dio en Flutter:
11//   interceptor.onRequest = (options, handler) {
12//     if (tokenValido) handler.next(options);
13//     else handler.reject(DioException.unauthorized);
14//   };
15
16use salvo::http::StatusCode;
17use salvo::prelude::*;
18use uuid::Uuid;
19
20use crate::services::AuthService;
21
22/// Middleware que verifica el JWT en el header Authorization.
23///
24/// # Flujo
25/// 1. Busca el header `Authorization: Bearer <token>`
26/// 2. Extrae el token (quita el prefijo "Bearer ")
27/// 3. Verifica el JWT con AuthService
28/// 4. Si es valido: inserta `id_usuario` en Depot → call_next
29/// 5. Si no: responde 401 → skip_rest
30#[handler]
31pub async fn auth_middleware(
32    req: &mut Request,
33    depot: &mut Depot,
34    res: &mut Response,
35    ctrl: &mut FlowCtrl,
36) {
37    // Obtenemos el AuthService del Depot (inyectado en main.rs)
38    let auth_service = match depot.obtain::<AuthService>() {
39        Ok(service) => service.clone(),
40        Err(_) => {
41            res.status_code(StatusCode::INTERNAL_SERVER_ERROR);
42            res.render(Json(serde_json::json!({
43                "error": "Error interno del servidor",
44                "codigo": 500
45            })));
46            ctrl.skip_rest();
47            return;
48        }
49    };
50
51    // Extraemos el token del header Authorization
52    let token = match req.header::<String>("authorization") {
53        Some(header) if header.starts_with("Bearer ") => header[7..].to_string(),
54        _ => {
55            res.status_code(StatusCode::UNAUTHORIZED);
56            res.render(Json(serde_json::json!({
57                "error": "No autenticado",
58                "codigo": 401
59            })));
60            ctrl.skip_rest();
61            return;
62        }
63    };
64
65    // Verificamos el JWT
66    match auth_service.verificar_access_token(&token) {
67        Ok(claims) => {
68            // Token valido: extraemos el UUID del usuario y lo metemos en Depot
69            if let Ok(id_usuario) = Uuid::parse_str(&claims.sub) {
70                depot.insert("id_usuario", id_usuario);
71                ctrl.call_next(req, depot, res).await;
72            } else {
73                res.status_code(StatusCode::UNAUTHORIZED);
74                res.render(Json(serde_json::json!({
75                    "error": "Token con formato invalido",
76                    "codigo": 401
77                })));
78                ctrl.skip_rest();
79            }
80        }
81        Err(_) => {
82            res.status_code(StatusCode::UNAUTHORIZED);
83            res.render(Json(serde_json::json!({
84                "error": "Token invalido o expirado",
85                "codigo": 401
86            })));
87            ctrl.skip_rest();
88        }
89    }
90}
