1// Modelo preparado para OAuth (Google, Apple) — fase 2.
2// Se permite dead_code porque la entidad existe en la BD
3// y sera usada cuando se implemente autenticacion con proveedores externos.
4
5use sea_orm::entity::prelude::*;
6use serde::{Deserialize, Serialize};
7
8#[allow(dead_code)]
9#[derive(Clone, Debug, PartialEq, DeriveEntityModel, Serialize, Deserialize)]
10#[sea_orm(table_name = "proveedores_autenticacion")]
11pub struct Model {
12    #[sea_orm(primary_key, auto_increment = false)]
13    pub id: Uuid,
14    pub id_usuario: Uuid,
15    pub proveedor: Option<String>,
16    pub identificador_proveedor: Option<String>,
17    pub email_proveedor: Option<String>,
18    /// JSONB en PostgreSQL → serde_json::Value en Rust
19    #[sea_orm(column_type = "JsonBinary")]
20    pub datos_proveedor: Option<serde_json::Value>,
21    pub creado_en: Option<DateTimeWithTimeZone>,
22    pub actualizado_en: Option<DateTimeWithTimeZone>,
23}
24
25#[allow(dead_code)]
26#[derive(Copy, Clone, Debug, EnumIter, DeriveRelation)]
27pub enum Relation {
28    #[sea_orm(
29        belongs_to = "super::usuario::Entity",
30        from = "Column::IdUsuario",
31        to = "super::usuario::Column::Id"
32    )]
33    Usuario,
34}
35
36impl Related<super::usuario::Entity> for Entity {
37    fn to() -> RelationDef {
38        Relation::Usuario.def()
39    }
40}
