1//! Handlers HTTP para endpoints de geografia.
2//!
3//! NO hay SQL aqui — los handlers llaman al repositorio.
4//! Los tipos Model de SeaORM son serializables a JSON directamente.
5//!
6//! Todos los handlers usan `#[endpoint]` para que Salvo genere
7//! documentacion OpenAPI automaticamente de cada ruta.
8//!
9//! # Por que `Json<serde_json::Value>` para respuestas de SeaORM
10//! Los modelos de SeaORM (`comunidad_autonoma::Model`, etc.) no implementan
11//! `ToSchema` de Salvo OAPI — implementarlo requeriria modificar los derives
12//! de `DeriveEntityModel`, lo que puede generar conflictos. En vez de eso,
13//! serializamos a `serde_json::Value` (JSON opaco) para la respuesta.
14//! OpenAPI documenta estos endpoints como que devuelven un objeto JSON generico.
15//! Los DTOs que controlamos directamente (auth, ofertas) si tienen `ToSchema`.
16
17use salvo::oapi::extract::{PathParam, QueryParam};
18use salvo::prelude::*;

20use crate::errors::AppError;

23/// GET /api/v1/comunidades — Listar todas las comunidades autonomas de Espana.
24///
25/// Devuelve las 17 CCAA + Ceuta + Melilla con datos del servicio de empleo regional
26/// (nombre, web, URL de sellado de demanda de empleo).
27///
28/// # Por que no necesita parametros de ruta ni query
29/// Este endpoint devuelve siempre la lista completa (son solo 19 registros estaticos).
30/// No necesita paginacion ni filtros porque el dataset es pequeno y no cambia.
31///
32/// # Por que `depot: &mut Depot` sigue siendo necesario
33/// OpenAPI ignora `depot` en la firma — no lo documenta como parametro.
34/// Pero el runtime de Salvo lo necesita para que el handler pueda acceder
35/// a los servicios/repositorios inyectados en main.rs con `affix_state::inject`.
36#[endpoint(tags("Geografia"))]
37pub async fn listar_comunidades(depot: &mut Depot) -> Result<Json<serde_json::Value>, AppError> {
38    let repo = depot
39        .obtain::<SeaGeografiaRepo>()
40        .map_err(|_| AppError::Internal("Repositorio de geografia no disponible".into()))?;
41
42    let comunidades = repo.listar_comunidades().await?;
43
44    Ok(Json(serde_json::to_value(comunidades).unwrap_or_default()))
45}

47/// GET /api/v1/comunidades/{id} — Obtener una comunidad autonoma por su ID.
48///
49/// # Por que `PathParam<i32>` en vez de `req.param::<i32>("id")`
50/// `PathParam<i32>` es un extractor tipado de Salvo OAPI. Salvo lo detecta
51/// en la firma de la funcion y lo documenta como parametro de ruta en OpenAPI
52/// con tipo `integer`. Con `req.param()`, Salvo no puede inferir el tipo
53/// ni incluirlo en la documentacion automatica.
54///
55/// # Por que `*id` (dereference)
56/// `id` es de tipo `PathParam<i32>`, no `i32` directamente.
57/// `PathParam<T>` implementa `Deref<Target = T>`, lo que permite usar `*id`
58/// para obtener el `i32` subyacente. Es como unwrap pero a nivel de tipos.
59#[endpoint(tags("Geografia"))]
60pub async fn obtener_comunidad(
61    id: PathParam<i32>,
62    depot: &mut Depot,
63) -> Result<Json<serde_json::Value>, AppError> {
64    let repo = depot
65        .obtain::<SeaGeografiaRepo>()
66        .map_err(|_| AppError::Internal("Repositorio de geografia no disponible".into()))?;
67
68    let comunidad = repo.obtener_comunidad(*id).await?;
69
70    Ok(Json(serde_json::to_value(comunidad).unwrap_or_default()))
71}

73/// GET /api/v1/provincias — Listar provincias con filtro opcional por comunidad.
74///
75/// # Parametros de query
76/// - `id_comunidad` (opcional): si se pasa, filtra las provincias de esa CCAA.
77///   Sin este parametro, devuelve las 52 provincias espanolas.
78///
79/// # Por que `QueryParam<i32, false>`
80/// `QueryParam<T, REQUIRED>` es el extractor de parametros de query URL de Salvo OAPI.
81/// El segundo parametro de tipo (`false`) indica que es OPCIONAL.
82/// `QueryParam<i32, true>` seria obligatorio.
83/// Salvo lo documenta en OpenAPI como `?id_comunidad=X` (query param).
84///
85/// # Por que `id_comunidad.into_inner()`
86/// `QueryParam<i32, false>` wrappea un `Option<i32>` (puede no venir en la URL).
87/// `.into_inner()` extrae el `Option<i32>` del wrapper para pasarlo al repositorio.
88#[endpoint(tags("Geografia"))]
89pub async fn listar_provincias(
90    id_comunidad: QueryParam<i32, false>,
91    depot: &mut Depot,
92) -> Result<Json<serde_json::Value>, AppError> {
93    let repo = depot
94        .obtain::<SeaGeografiaRepo>()
95        .map_err(|_| AppError::Internal("Repositorio de geografia no disponible".into()))?;
96
97    let provincias = repo.listar_provincias(id_comunidad.into_inner()).await?;
98
99    Ok(Json(serde_json::to_value(provincias).unwrap_or_default()))
100}
