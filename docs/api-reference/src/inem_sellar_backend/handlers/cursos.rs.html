1//! CRUD de cursos de formacion.
2//!
3//! GET = publico, POST/PUT/DELETE = requiere autenticacion JWT.
4//!
5//! Los handlers usan `#[endpoint]` (en lugar del antiguo `#[handler]`) para
6//! generacion automatica de documentacion OpenAPI. La diferencia clave es que
7//! `#[endpoint]` analiza los tipos de los parametros en tiempo de compilacion
8//! y construye el esquema OpenAPI sin necesidad de escribirlo a mano.
9//!
10//! Los endpoints protegidos llevan `security(("bearer_auth" = []))`, lo que
11//! indica a Swagger UI que muestren un candado y permitan introducir un JWT
12//! para probar el endpoint directamente desde la documentacion.
13

25/// Body para crear un nuevo curso de formacion.
26///
27/// # Por que `ToSchema`
28/// El derive `ToSchema` le indica a Salvo que genere el esquema JSON Schema
29/// de este struct automaticamente. Asi Swagger UI puede mostrar un ejemplo
30/// del body con todos los campos y sus tipos — equivalente a documentacion
31/// de parametros en Flutter/Dart pero generada 100% desde el codigo Rust.
32///
33/// # Por que todos los campos son `Option<T>`
34/// En Rust, un campo `Option<String>` puede ser `Some("valor")` o `None`
35/// (ausente). Esto modela publicaciones parciales: el usuario no tiene por
36/// que rellenar todos los campos al crear un curso.
37/// En Dart usarias `String?` — aqui es exactamente lo mismo.
38#[derive(Deserialize, ToSchema)]
39pub struct CrearCursoRequest {
40    /// Titulo del curso. Ej: "Introduccion a la Ciberseguridad".
41    pub titulo: Option<String>,
42    /// Descripcion breve del curso (resumen de una o dos frases).
43    pub descripcion: Option<String>,
44    /// Contenido detallado o temario del curso (puede ser largo).
45    pub contenido: Option<String>,
46    /// Enlace a la pagina web del curso o entidad organizadora.
47    pub web: Option<String>,
48    /// URL de una imagen representativa del curso.
49    pub imagen_url: Option<String>,
50    /// Duracion total del curso en horas. Ej: 40.
51    pub duracion_horas: Option<i32>,
52    /// Fecha de inicio en formato ISO 8601: "YYYY-MM-DD". Ej: "2026-05-01".
53    pub fecha_inicio: Option<String>,
54    /// Fecha de fin en formato ISO 8601: "YYYY-MM-DD". Ej: "2026-06-30".
55    pub fecha_fin: Option<String>,
56    /// Indica si el curso esta homologado oficialmente por la administracion.
57    pub curso_homologado: Option<bool>,
58    /// Telefono de contacto para informacion sobre el curso.
59    pub telefono_contacto: Option<String>,
60    /// Email de contacto para inscripciones o consultas.
61    pub email_contacto: Option<String>,
62    /// Lista de IDs de provincias donde se imparte el curso (codigos INE).
63    /// Lista vacia = curso disponible en toda Espana (nacional).
64    ///
65    /// # Por que `#[serde(default)]`
66    /// Si el cliente no envia el campo `provincias` en el JSON, serde usara
67    /// el valor por defecto de `Vec<i32>`, que es un vector vacio `[]`.
68    /// Sin esta anotacion, la deserializacion fallaria si falta el campo.
69    #[serde(default)]
70    pub provincias: Vec<i32>,
71}

73/// Body para actualizar un curso existente.
74///
75/// Solo el autor original puede actualizar el curso.
76/// Unicamente se modifican los campos que se incluyan en el JSON
77/// (partial update / PATCH semantics sobre un endpoint PUT).
78///
79/// # Por que `Option<Vec<i32>>` para provincias
80/// Distingue tres estados posibles:
81/// - Campo ausente (`None`): no tocar las provincias actuales.
82/// - Campo presente con valores (`Some([1, 2])`): reemplazar todas las provincias.
83/// - Campo presente y vacio (`Some([])`): eliminar todas las provincias.
84///
85/// Un simple `Vec<i32>` no podria representar "no tocar" vs "eliminar todas".
86#[derive(Deserialize, ToSchema)]
87pub struct ActualizarCursoRequest {
88    /// Nuevo titulo del curso (si no se envia, el titulo no cambia).
89    pub titulo: Option<String>,
90    /// Nueva descripcion breve.
91    pub descripcion: Option<String>,
92    /// Nuevo contenido detallado.
93    pub contenido: Option<String>,
94    /// Nueva URL de la pagina web del curso.
95    pub web: Option<String>,
96    /// Nueva URL de imagen.
97    pub imagen_url: Option<String>,
98    /// Nueva duracion en horas.
99    pub duracion_horas: Option<i32>,
100    /// Nueva fecha de inicio "YYYY-MM-DD".
101    pub fecha_inicio: Option<String>,
102    /// Nueva fecha de fin "YYYY-MM-DD".
103    pub fecha_fin: Option<String>,
104    /// Nuevo valor de homologacion.
105    pub curso_homologado: Option<bool>,
106    /// Nuevo telefono de contacto.
107    pub telefono_contacto: Option<String>,
108    /// Nuevo email de contacto.
109    pub email_contacto: Option<String>,
110    /// Nuevas provincias asociadas. Si se envia, reemplaza las existentes.
111    /// No enviar este campo = no tocar provincias. Enviar `[]` = eliminar todas.
112    pub provincias: Option<Vec<i32>>,
113}

115/// Respuesta paginada del listado de cursos.
116///
117/// # Por que `Vec<serde_json::Value>` en lugar de `Vec<curso::Model>`
118/// Los cursos pueden tener datos agregados dinamicamente (provincias, voto
119/// del usuario actual) que no estan en el Model base de SeaORM. Usando
120/// `serde_json::Value` tenemos flexibilidad para enriquecer la respuesta
121/// sin crear structs adicionales. La contrapartida es que OpenAPI no puede
122/// documentar la estructura interna de cada curso — limitacion conocida y aceptada.
123#[derive(Serialize, ToSchema)]
124pub struct ListaCursosResponse {
125    /// Lista de cursos de la pagina actual.
126    pub cursos: Vec<serde_json::Value>,
127    /// Total de cursos que coinciden con el filtro (para calcular el numero de paginas).
128    pub total: u64,
129    /// Numero de pagina actual, base 1.
130    pub pagina: u64,
131    /// Numero de cursos incluidos por pagina.
132    pub por_pagina: u64,
133}

137/// Convierte un string con formato "YYYY-MM-DD" a `chrono::NaiveDate`.
138///
139/// # Por que una funcion helper separada
140/// La logica de parseo de fechas se repite en `crear_curso` y `actualizar_curso`.
141/// Extraerla a una funcion evita duplicacion (principio DRY) y centraliza
142/// el mensaje de error para que sea consistente en ambos endpoints.
143///
144/// `NaiveDate` es la representacion de chrono para una fecha sin zona horaria
145/// (solo dia, mes, anho). Es lo que espera SeaORM para columnas `DATE` en PostgreSQL.
146fn parsear_fecha(s: &str) -> Result<chrono::NaiveDate, AppError> {
147    chrono::NaiveDate::parse_from_str(s, "%Y-%m-%d").map_err(|_| {
148        AppError::BadRequest(format!("Fecha invalida: {s}. Formato esperado: YYYY-MM-DD"))
149    })
150}

154/// GET /api/v1/cursos — Listar cursos activos con paginacion y filtro opcional por provincia.
155///
156/// Endpoint publico — no requiere autenticacion.
157///
158/// # Por que `QueryParam<i32, false>` y no `req.query::<i32>(...)`
159/// `QueryParam<T, false>` es un extractor tipado de Salvo que:
160/// 1. Deserializa el query param automaticamente al tipo `T`.
161/// 2. El segundo argumento `false` indica que el parametro es OPCIONAL
162///    (si fuera `true` seria obligatorio y devolveria 400 si falta).
163/// 3. Salvo lo incluye en el esquema OpenAPI para que Swagger UI muestre
164///    los parametros de query con sus tipos.
165///
166/// # Parametros de query (todos opcionales)
167/// - `id_provincia`: filtrar por provincia (codigo INE 1-52)
168/// - `pagina`: numero de pagina, base 1 (default: 1)
169/// - `por_pagina`: resultados por pagina (default: 20)
170#[endpoint(tags("Cursos"))]
171pub async fn listar_cursos(
172    id_provincia: QueryParam<i32, false>,
173    pagina: QueryParam<u64, false>,
174    por_pagina: QueryParam<u64, false>,
175    depot: &mut Depot,
176) -> Result<Json<ListaCursosResponse>, AppError> {
177    // `into_inner()` extrae el valor del wrapper QueryParam.
178    // `unwrap_or` proporciona el valor por defecto cuando el param no se envio.
179    let pagina_val = pagina.into_inner().unwrap_or(1);
180    let por_pagina_val = por_pagina.into_inner().unwrap_or(20);
181
182    let repo = depot
183        .obtain::<SeaCursoRepo>()
184        .map_err(|_| AppError::Internal("CursoRepo no disponible".into()))?
185        .clone();
186
187    let (cursos, total) = repo
188        .listar_cursos(id_provincia.into_inner(), pagina_val, por_pagina_val)
189        .await?;
190
191    // Convertimos cada Model de SeaORM a serde_json::Value para la respuesta.
192    // `unwrap_or_default()` devuelve `null` si la serializacion falla (nunca deberia).
193    let cursos_json: Vec<serde_json::Value> = cursos
194        .into_iter()
195        .map(|c| serde_json::to_value(c).unwrap_or_default())
196        .collect();
197
198    Ok(Json(ListaCursosResponse {
199        cursos: cursos_json,
200        total,
201        pagina: pagina_val,
202        por_pagina: por_pagina_val,
203    }))
204}

206/// GET /api/v1/cursos/{id} — Obtener un curso por su UUID.
207///
208/// Endpoint publico — no requiere autenticacion.
209///
210/// # Por que `PathParam<String>` y no directamente `PathParam<Uuid>`
211/// Salvo puede extraer el path param como `String` de forma segura y sin
212/// dependencias extra. Luego parseamos el UUID manualmente con `Uuid::parse_str`
213/// para producir un error 400 descriptivo si el formato no es correcto.
214/// Esto da mas control sobre el mensaje de error que dejar que Salvo falle internamente.
215#[endpoint(tags("Cursos"))]
216pub async fn obtener_curso(
217    id: PathParam<String>,
218    depot: &mut Depot,
219) -> Result<Json<serde_json::Value>, AppError> {
220    // El operador `?` propaga el error si el UUID no es valido, retornando 400 Bad Request.
221    let uuid = Uuid::parse_str(&id)
222        .map_err(|_| AppError::BadRequest("ID de curso no es un UUID valido".into()))?;
223
224    let repo = depot
225        .obtain::<SeaCursoRepo>()
226        .map_err(|_| AppError::Internal("CursoRepo no disponible".into()))?
227        .clone();
228
229    let curso = repo.obtener_curso(uuid).await?;
230
231    Ok(Json(serde_json::to_value(curso).unwrap_or_default()))
232}

234/// POST /api/v1/cursos — Crear un nuevo curso (requiere autenticacion JWT).
235///
236/// El `id_autor` se extrae del JWT en el middleware de autenticacion —
237/// nunca se acepta del body para evitar que un usuario publique contenido
238/// en nombre de otro.
239///
240/// # Por que `JsonBody<CrearCursoRequest>`
241/// `JsonBody<T>` es el extractor tipado de Salvo para el body de la peticion.
242/// Deserializa el JSON automaticamente al tipo `T` y genera el esquema en OpenAPI.
243/// Equivale a `req.parse_json::<T>().await` del antiguo `#[handler]`, pero
244/// con documentacion automatica incluida.
245///
246/// # Por que `security(("bearer_auth" = []))`
247/// Esta anotacion indica a OpenAPI que este endpoint requiere autenticacion JWT.
248/// Swagger UI mostrara un candado y permitira introducir el token Bearer para
249/// probar el endpoint directamente desde la documentacion.
250#[endpoint(tags("Cursos"), security(("bearer_auth" = [])))]
251pub async fn crear_curso(
252    body: JsonBody<CrearCursoRequest>,
253    depot: &mut Depot,
254) -> Result<Json<serde_json::Value>, AppError> {
255    // `depot.get::<Uuid>("id_usuario")` recupera el UUID inyectado por auth_middleware.
256    // El `*` dereferencia el `&Uuid` a `Uuid` (Copy trait — no necesita clone).
257    let id_autor = *depot
258        .get::<Uuid>("id_usuario")
259        .map_err(|_| AppError::Unauthorized)?;
260
261    let repo = depot
262        .obtain::<SeaCursoRepo>()
263        .map_err(|_| AppError::Internal("CursoRepo no disponible".into()))?
264        .clone();
265
266    // Construimos el DTO (Data Transfer Object) que el repositorio espera.
267    // Las fechas vienen como String en el JSON y las convertimos a NaiveDate aqui,
268    // antes de pasarlas al repositorio que solo conoce tipos de dominio.
269    //
270    // `as_deref()` convierte `Option<String>` a `Option<&str>` sin consumir el valor.
271    // `map(parsear_fecha)` aplica la conversion a cada `&str` si existe.
272    // `transpose()` convierte `Option<Result<T, E>>` a `Result<Option<T>, E>`,
273    // propagando el error con `?` si la fecha es invalida.
274    let dto = CrearCursoDto {
275        titulo: body.titulo.clone(),
276        descripcion: body.descripcion.clone(),
277        contenido: body.contenido.clone(),
278        web: body.web.clone(),
279        imagen_url: body.imagen_url.clone(),
280        duracion_horas: body.duracion_horas,
281        fecha_inicio: body
282            .fecha_inicio
283            .as_deref()
284            .map(parsear_fecha)
285            .transpose()?,
286        fecha_fin: body.fecha_fin.as_deref().map(parsear_fecha).transpose()?,
287        curso_homologado: body.curso_homologado,
288        telefono_contacto: body.telefono_contacto.clone(),
289        email_contacto: body.email_contacto.clone(),
290        provincias: body.provincias.clone(),
291    };
292
293    let curso = repo.crear_curso(id_autor, dto).await?;
294
295    Ok(Json(serde_json::to_value(curso).unwrap_or_default()))
296}

298/// PUT /api/v1/cursos/{id} — Actualizar un curso (solo el autor original).
299///
300/// Verifica que el usuario autenticado sea el autor del curso antes de
301/// aplicar cualquier cambio. Si no lo es, devuelve 403 Forbidden.
302///
303/// Solo se modifican los campos que se incluyan en el JSON enviado.
304/// El campo `provincias` tiene semantica especial:
305/// - Ausente: no se tocan las provincias actuales.
306/// - `[]`: se eliminan todas las provincias (curso nacional).
307/// - `[1, 2, 3]`: se reemplazan las provincias por las indicadas.
308#[endpoint(tags("Cursos"), security(("bearer_auth" = [])))]
309pub async fn actualizar_curso(
310    id: PathParam<String>,
311    body: JsonBody<ActualizarCursoRequest>,
312    depot: &mut Depot,
313) -> Result<Json<serde_json::Value>, AppError> {
314    let id_usuario = *depot
315        .get::<Uuid>("id_usuario")
316        .map_err(|_| AppError::Unauthorized)?;
317
318    let uuid = Uuid::parse_str(&id)
319        .map_err(|_| AppError::BadRequest("ID de curso no es un UUID valido".into()))?;
320
321    let repo = depot
322        .obtain::<SeaCursoRepo>()
323        .map_err(|_| AppError::Internal("CursoRepo no disponible".into()))?
324        .clone();
325
326    // Verificar autoria: solo el creador del curso puede modificarlo.
327    // `id_autor` es `Option<Uuid>` en el modelo — comparamos con `Some(id_usuario)`.
328    let curso_existente = repo.obtener_curso(uuid).await?;
329    if curso_existente.id_autor != Some(id_usuario) {
330        return Err(AppError::Forbidden);
331    }
332
333    let dto = ActualizarCursoDto {
334        titulo: body.titulo.clone(),
335        descripcion: body.descripcion.clone(),
336        contenido: body.contenido.clone(),
337        web: body.web.clone(),
338        imagen_url: body.imagen_url.clone(),
339        duracion_horas: body.duracion_horas,
340        fecha_inicio: body
341            .fecha_inicio
342            .as_deref()
343            .map(parsear_fecha)
344            .transpose()?,
345        fecha_fin: body.fecha_fin.as_deref().map(parsear_fecha).transpose()?,
346        curso_homologado: body.curso_homologado,
347        telefono_contacto: body.telefono_contacto.clone(),
348        email_contacto: body.email_contacto.clone(),
349        // Pasamos las provincias tal cual vienen del body:
350        // - None si el campo no estaba en el JSON => el repo no toca las provincias.
351        // - Some(vec) si el campo estaba => el repo reemplaza las provincias.
352        provincias: body.provincias.clone(),
353    };
354
355    let curso = repo.actualizar_curso(uuid, dto).await?;
356
357    Ok(Json(serde_json::to_value(curso).unwrap_or_default()))
358}

360/// DELETE /api/v1/cursos/{id} — Eliminar un curso (solo el autor original).
361///
362/// Elimina fisicamente el curso de la base de datos junto con todas sus
363/// relaciones de provincia (gracias a `ON DELETE CASCADE` en el esquema SQL).
364///
365/// La verificacion de autoria se realiza ANTES del borrado para evitar
366/// eliminar contenido de otros usuarios por error.
367#[endpoint(tags("Cursos"), security(("bearer_auth" = [])))]
368pub async fn eliminar_curso(
369    id: PathParam<String>,
370    depot: &mut Depot,
371) -> Result<Json<serde_json::Value>, AppError> {
372    let id_usuario = *depot
373        .get::<Uuid>("id_usuario")
374        .map_err(|_| AppError::Unauthorized)?;
375
376    let uuid = Uuid::parse_str(&id)
377        .map_err(|_| AppError::BadRequest("ID de curso no es un UUID valido".into()))?;
378
379    let repo = depot
380        .obtain::<SeaCursoRepo>()
381        .map_err(|_| AppError::Internal("CursoRepo no disponible".into()))?
382        .clone();
383
384    // Comprobar autoria antes de eliminar.
385    let curso = repo.obtener_curso(uuid).await?;
386    if curso.id_autor != Some(id_usuario) {
387        return Err(AppError::Forbidden);
388    }
389
390    repo.eliminar_curso(uuid).await?;
391
392    Ok(Json(
393        serde_json::json!({"mensaje": "Curso eliminado correctamente"}),
394    ))
