1//! Handlers de reportes: crear, listar y procesar reportes de contenido.
2//!
3//! Un reporte es la forma en que los usuarios notifican contenido inapropiado
4//! (spam, incorrecto, duplicado, etc.) a los moderadores. La tabla `reportes`
5//! es polimorfica — un mismo reporte puede referirse a una oferta, un consejo
6//! o un curso, identificado por el par `(tipo_contenido, id_contenido)`.
7//!
8//! Flujo de moderacion:
9//!   1. Usuario crea un reporte (`crear_reporte`) — estado: `pendiente`
10//!   2. Admin lista reportes pendientes (`listar_reportes_pendientes`)
11//!   3. Admin acepta o rechaza (`procesar_reporte`) — estado: `aceptado` | `rechazado`
12

25/// Convierte un `String` recibido del cliente en el enum `TipoContenido`.
26///
27/// # Por que `String` en el DTO en lugar del enum directamente
28/// Los enums de SeaORM (`DeriveActiveEnum`) no implementan el trait `ToSchema`
29/// de Salvo OAPI. Sin `ToSchema`, `#[endpoint]` no puede generar la
30/// documentacion OpenAPI del campo. Solución: recibir como `String` y
31/// convertir aqui. Es el mismo patron que usa `votos.rs`.
32fn parsear_tipo_contenido(s: &str) -> Result<TipoContenido, AppError> {
33    match s {
34        "oferta" => Ok(TipoContenido::Oferta),
35        "consejo" => Ok(TipoContenido::Consejo),
36        "curso" => Ok(TipoContenido::Curso),
37        otro => Err(AppError::BadRequest(format!(
38            "tipo_contenido invalido: '{otro}'. Valores validos: oferta, consejo, curso"
39        ))),
40    }
41}
42
43/// Convierte un `String` recibido del cliente en el enum `MotivoReporte`.
44///
45/// `match` en Rust es exhaustivo — el compilador exige que cubras todos los
46/// casos posibles. Aqui el caso `otro` actua como el "else" que captura
47/// cualquier string no reconocido y devuelve un error descriptivo.
48fn parsear_motivo(s: &str) -> Result<MotivoReporte, AppError> {
49    match s {
50        "spam" => Ok(MotivoReporte::Spam),
51        "inapropiado" => Ok(MotivoReporte::Inapropiado),
52        "desactualizado" => Ok(MotivoReporte::Desactualizado),
53        "incorrecto" => Ok(MotivoReporte::Incorrecto),
54        "duplicado" => Ok(MotivoReporte::Duplicado),
55        "otro" => Ok(MotivoReporte::Otro),
56        otro => Err(AppError::BadRequest(format!(
57            "motivo invalido: '{otro}'. Valores validos: spam, inapropiado, desactualizado, incorrecto, duplicado, otro"
58        ))),
59    }
60}

64/// Body para POST /api/v1/reportes — reportar un contenido.
65///
66/// # Por que `ToSchema`
67/// Permite que Salvo OAPI documente automaticamente la estructura del body
68/// en Swagger UI. Sin el, el generador de OpenAPI no puede describir los
69/// campos esperados para este endpoint.
70///
71/// # Por que `Deserialize`
72/// `Deserialize` (de serde) genera el codigo para convertir el JSON de la
73/// peticion en este struct de Rust. Es el equivalente del `fromJson()` en
74/// Dart, pero generado por una macro en tiempo de compilacion.
75#[derive(Deserialize, ToSchema)]
76pub struct CrearReporteRequest {
77    /// Tipo de contenido reportado. Valores: `"oferta"`, `"consejo"`, `"curso"`.
78    pub tipo_contenido: String,
79    /// UUID del contenido que se reporta.
80    pub id_contenido: Uuid,
81    /// Motivo del reporte. Valores: `"spam"`, `"inapropiado"`, `"desactualizado"`,
82    /// `"incorrecto"`, `"duplicado"`, `"otro"`.
83    pub motivo: String,
84    /// Explicacion adicional del reporte (opcional).
85    pub detalle_motivo: Option<String>,
86}

88/// Body para PUT /api/v1/reportes/{id} — procesar un reporte (aceptar o rechazar).

108/// POST /api/v1/reportes — Reportar un contenido.
109///
110/// Un usuario solo puede reportar el mismo contenido una vez — la BD tiene
111/// una restriccion `UNIQUE (tipo_contenido, id_contenido, id_reportero)`.
112/// Si intenta reportar dos veces el mismo contenido, recibe un error 409 Conflict.
113///
114/// El `id_reportero` se extrae del JWT, no del body, para evitar suplantacion.
115///
116/// # Por que `#[endpoint]` y `JsonBody<T>`
117/// `#[endpoint]` genera metadata OpenAPI ademas de registrar el handler HTTP.
118/// `JsonBody<T>` es el extractor tipado que Salvo OAPI necesita para documentar
119/// el body en Swagger. Ver `auth.rs` para una explicacion mas detallada del patron.
120#[endpoint(tags("Reportes"), security(("bearer_auth" = [])))]
121pub async fn crear_reporte(
122    body: JsonBody<CrearReporteRequest>,
123    depot: &mut Depot,
124) -> Result<Json<serde_json::Value>, AppError> {
125    // Extraer el UUID del reportero del Depot (inyectado por el middleware JWT).
126    // `depot.get::<T>(key)` devuelve `Result<&T, _>`. El `*` desreferencia
127    // la referencia para obtener el `Uuid` por valor (es `Copy`, no `Clone`).
128    let id_reportero = *depot
129        .get::<Uuid>("id_usuario")
130        .map_err(|_| AppError::Unauthorized)?;
131
132    // Parsear los campos String a sus enums correspondientes.
133    // Si cualquiera falla, el operador `?` propaga el AppError::BadRequest
134    // al caller (Salvo), que lo convierte en una respuesta 400.
135    let tipo = parsear_tipo_contenido(&body.tipo_contenido)?;
136    let motivo = parsear_motivo(&body.motivo)?;
137
138    let repo = depot
139        .obtain::<SeaReporteRepo>()
140        .map_err(|_| AppError::Internal("ReporteRepo no disponible".into()))?
141        .clone();
142
143    let dto = CrearReporteDto {
144        tipo_contenido: tipo,
145        id_contenido: body.id_contenido,
146        motivo,
147        detalle_motivo: body.detalle_motivo.clone(),
148    };
149
150    let reporte = repo.crear_reporte(id_reportero, dto).await?;
151
152    Ok(Json(serde_json::to_value(reporte).unwrap_or_default()))
153}

155/// GET /api/v1/reportes/pendientes — Listar reportes pendientes de revision.
156///
157/// Endpoint de administracion — devuelve todos los reportes cuyo estado
158/// es `pendiente`. En el futuro se anadira verificacion de rol admin.

164    // Verificar autenticacion — el middleware ya lo comprueba, pero este

166    let _id_usuario = depot
167        .get::<Uuid>("id_usuario")
168        .map_err(|_| AppError::Unauthorized)?;
169
170    let repo = depot

180/// PUT /api/v1/reportes/{id} — Procesar un reporte: aceptar o rechazar.
181///
182/// Registra quien proceso el reporte (`id_procesador` del JWT) y cuando
183/// (`procesado_en` se guarda en el repositorio). El estado del reporte pasa
184/// de `pendiente` a `aceptado` o `rechazado` segun el campo `aceptar`.
185///
186/// # Por que `PathParam<String>` en vez de `PathParam<Uuid>`
187/// Salvo OAPI puede documentar `PathParam<String>` sin problemas.
188/// Convertimos a `Uuid` manualmente para un mensaje de error claro
189/// si el formato es invalido, en lugar de dejar que Salvo devuelva
190/// un error generico de parseo.
191#[endpoint(tags("Reportes"), security(("bearer_auth" = [])))]
192pub async fn procesar_reporte(
193    id: PathParam<String>,
194    body: JsonBody<ProcesarReporteRequest>,
195    depot: &mut Depot,
196) -> Result<Json<serde_json::Value>, AppError> {
197    // El procesador es el usuario autenticado (admin en el futuro).
198    let id_procesador = *depot
199        .get::<Uuid>("id_usuario")
200        .map_err(|_| AppError::Unauthorized)?;
201
202    // `into_inner()` extrae el `String` del wrapper `PathParam<String>`.
203    // Es el patron idiomatico de Salvo para acceder al valor del extractor.
204    let uuid = Uuid::parse_str(&id)
205        .map_err(|_| AppError::BadRequest("ID de reporte no es un UUID valido".into()))?;
206
207    let repo = depot
208        .obtain::<SeaReporteRepo>()
209        .map_err(|_| AppError::Internal("ReporteRepo no disponible".into()))?
210        .clone();
211
212    let reporte = repo
213        .procesar_reporte(uuid, id_procesador, body.aceptar)
214        .await?;
215
216    Ok(Json(serde_json::to_value(reporte).unwrap_or_default()))
217}
