1//! Handlers de autenticacion: firebase (handshake unico), refrescar, logout.
2//!
3//! El cliente delega TODO el login en FirebaseAuth (Google, email/password,
4//! anonimo) y entrega el Firebase ID Token en `/auth/firebase`. El backend
5//! lo valida contra los JWKS de Google y emite sus propios tokens (JWT
6//! HS256 + refresh opaco), que son los que usan el resto de endpoints.
7//!
8//! Cada handler coordina servicio + repositorio, pero NO contiene
9//! logica de criptografia ni acceso a BD directamente.
10//!
11//! Los handlers usan `#[endpoint]` (en vez de `#[handler]`) para que
12//! Salvo pueda generar la documentacion OpenAPI automaticamente.
13

59/// Respuesta del handshake `/auth/firebase` — contiene tokens propios
60/// (HS256) + datos basicos del usuario.
61///
62/// # Por que `Serialize` y `ToSchema`
63/// `Serialize` (de serde) convierte el struct a JSON para la respuesta HTTP.
64/// `ToSchema` permite que OpenAPI documente la estructura de la respuesta.
65/// Juntos, forman el contrato completo: implementacion + documentacion.
66#[derive(Serialize, ToSchema)]
67pub struct AuthResponse {
68    /// Token JWT de acceso. Corta duracion (15 min por defecto).
69    /// Incluirlo en `Authorization: Bearer <token>` en cada peticion protegida.
70    pub access_token: String,
71    /// Token de refresco. Larga duracion (30 dias). Guardar de forma segura.
72    pub refresh_token: String,
73    /// Datos basicos del usuario recien autenticado.
74    pub usuario: UsuarioResponse,
75}
76
77/// Datos publicos del usuario (sin hash_contrasena).
78///
79/// # Por que campos nullable
80/// Esta struct la consumen los 4 sub-flujos del handshake Firebase:
81///   - `google.com`: tiene email, nombre y avatar de Google.
82///   - `apple.com`: tiene email (incluso si es relay
83///     `@privaterelay.appleid.com` por "Hide my email"); `name`/`picture`
84///     solo en el PRIMER login (Apple solo expone esos campos una vez).
85///   - `password` (Firebase Email/Password): tiene email; name/picture
86///     vienen del Firebase profile si el cliente los puso, opcionales.
87///   - `anonymous`: no tiene email ni nombre.
88///
89/// Todos los campos identitarios son `Option` para que el cliente reciba
90/// `null` en JSON y pueda tomar decisiones de UI sin romperse.
91#[derive(Serialize, ToSchema)]
92pub struct UsuarioResponse {
93    /// UUID del usuario en la base de datos.
94    pub id: Uuid,
95    /// Email del usuario. `None` para anonimos y para providers sin email.
96    pub email: Option<String>,
97    /// Nombre visible publicamente. `None` si no se proporciono.
98    pub nombre_visible: Option<String>,
99    /// URL del avatar. Procede del proveedor (ej: foto de Google) o del
100    /// perfil que el propio usuario subio. `None` si no hay foto.
101    pub url_avatar: Option<String>,
102    /// `sign_in_provider` literal de Firebase para ESTE login:
103    /// `"google.com"` | `"apple.com"` | `"password"` | `"anonymous"`. El
104    /// cliente lo usa para decidir que pantalla de re-login mostrar si
105    /// hace falta.
106    pub proveedor: Option<String>,
107    /// `true` si el usuario es anonimo. El cliente lo usa para mostrar el
108    /// flujo "completar registro".
109    pub anonimo: bool,
110    /// `true` si el proveedor (Google, etc.) verifico el email. `None` si no
111    /// aplica (anonimos, email/password sin verificacion).
112    pub email_verificado: Option<bool>,
113}

133/// Construye un `UsuarioResponse` a partir del modelo de BD y el contexto
134/// del login en curso (proveedor, flag anonimo, email verificado).
135///
136/// Centraliza la transformacion para que cualquier nuevo flujo de login
137/// devuelva un response consistente con los demas. Los handlers solo
138/// pasan el contexto que conocen (proveedor del login actual, etc.).
139fn construir_usuario_response(
140    user: &usuario::Model,
141    proveedor: &str,
142    anonimo: bool,
143    email_verificado: Option<bool>,
144) -> UsuarioResponse {
145    UsuarioResponse {
146        id: user.id,
147        email: user.email.clone(),
148        nombre_visible: user.nombre_visible.clone(),
149        url_avatar: user.url_avatar.clone(),
150        proveedor: Some(proveedor.to_string()),
151        anonimo,
152        email_verificado,
153    }
154}

156/// Construye el `datos_proveedor` JSONB para una fila Firebase a partir
157/// de los claims verificados del ID Token.
158///
159/// # Que se guarda
160/// Solo claims publicos (no la firma) que tengan utilidad operativa:
161///   - `sign_in_provider`: distingue google.com / apple.com / etc.
162///   - `email_verified`: para auditar account-linking en disputas.
163///   - `name`, `picture`: snapshot del perfil al momento del login.
164///   - `identities`: util si en el futuro un mismo usuario vincula
165///     varios providers (Google + Apple) — el `firebase_uid` no cambia
166///     pero las identidades si.
167///   - `auth_time`: cuando el usuario realmente autentico (puede ser
168///     anterior al `iat` si refresco con un session cookie).
169///   - `tenant`: para multitenancy (no usado en InemSellar pero lo
170///     guardamos por si acaso).
171///
172/// # Que NO se guarda
173/// El `id_token` original NUNCA. Es un secreto de corta vida que no
174/// debe persistir. Si se filtrara la BD, no se podria suplantar al
175/// usuario con estos datos solos.
176fn construir_datos_proveedor_firebase(claims: &FirebaseClaims) -> serde_json::Value {
177    serde_json::json!({
178        "sign_in_provider": claims.firebase.sign_in_provider,
179        "email_verified": claims.email_verified,
180        "name": claims.name,
181        "picture": claims.picture,
182        "identities": claims.firebase.identities,
183        "auth_time": claims.auth_time,
184        "tenant": claims.firebase.tenant,
185    })
186}

188/// Lee de forma defensiva el header `User-Agent` de la peticion para
189/// usarlo como `informacion_dispositivo` al guardar el refresh token.
190///
191/// Devuelve `None` si el header esta ausente o no es ASCII parseable.
192fn extraer_user_agent(req: &Request) -> Option<String> {
193    req.header::<String>("user-agent")
194        .filter(|s| !s.trim().is_empty())
195}

199/// POST /api/v1/auth/refrescar — Obtener nuevos tokens con un refresh token.
200///
201/// # Rotacion de tokens
202/// Al refrescar, el refresh token anterior se revoca y se emite uno nuevo.
203/// Esto limita el dano si un refresh token es comprometido: el atacante
204/// solo tiene una ventana corta antes de que el token se invalide.
205#[endpoint(tags("Auth"))]
206pub async fn refrescar(
207    req: &mut Request,
208    body: JsonBody<RefrescarRequest>,
209    depot: &mut Depot,
210) -> Result<Json<TokenResponse>, AppError> {
211    let auth_service = depot
212        .obtain::<AuthService>()
213        .map_err(|_| AppError::Internal("AuthService no disponible".into()))?
214        .clone();
215    let auth_repo = depot
216        .obtain::<SeaAuthRepo>()
217        .map_err(|_| AppError::Internal("AuthRepo no disponible".into()))?
218        .clone();
219    let proveedor_repo = depot
220        .obtain::<SeaProveedorAutenticacionRepo>()
221        .map_err(|_| AppError::Internal("ProveedorAutenticacionRepo no disponible".into()))?
222        .clone();
223
224    // Hashear el refresh token recibido y buscarlo en BD
225    let hash = auth_service.hashear_refresh_token(&body.refresh_token);
226    let token_db = auth_repo
227        .buscar_refresh_token_por_hash(&hash)
228        .await?
229        .ok_or(AppError::Unauthorized)?;
230
231    // Verificar que no ha expirado
232    if let Some(expira) = token_db.expira_en
233        && expira < Utc::now().fixed_offset()
234    {
235        return Err(AppError::Unauthorized);
236    }
237
238    // Revocar el token anterior (rotacion)
239    auth_repo.revocar_refresh_token(token_db.id).await?;
240
241    // Preservar el flag `anonimo` al rotar: consultamos si el usuario tiene
242    // una identidad `proveedor='anonymous'`. Asi un anonimo que refresca
243    // sigue siendo anonimo en el nuevo access_token.
244    let es_anonimo = proveedor_repo.es_anonimo(token_db.id_usuario).await?;
245
246    // Generar nuevo par de tokens manteniendo el flag.
247    let access_token =
248        auth_service.generar_access_token_con_flag(token_db.id_usuario, es_anonimo)?;
249    let new_refresh_raw = auth_service.generar_refresh_token();
250    let new_refresh_hash = auth_service.hashear_refresh_token(&new_refresh_raw);
251
252    let expira = (Utc::now() + Duration::days(30)).fixed_offset();
253    let user_agent = extraer_user_agent(req);
254    auth_repo
255        .guardar_refresh_token(
256            token_db.id_usuario,
257            &new_refresh_hash,
258            user_agent.as_deref(),
259            expira,
260        )
261        .await?;
262
263    Ok(Json(TokenResponse {
264        access_token,
265        refresh_token: new_refresh_raw,
266    }))
267}

269/// POST /api/v1/auth/logout — Cerrar sesion (revocar refresh token).
270/// Requiere autenticacion (access token en header Authorization).
271///
272/// # Por que `security(("bearer_auth" = []))`
273/// Esta anotacion le dice a OpenAPI que este endpoint requiere el esquema
274/// de seguridad `bearer_auth` definido en main.rs (JWT Bearer token).
275/// Swagger UI mostrara un candado y permitira al usuario introducir su token
276/// para probar el endpoint directamente desde la documentacion.
277#[endpoint(tags("Auth"), security(("bearer_auth" = [])))]
278pub async fn logout(
279    body: JsonBody<LogoutRequest>,
280    depot: &mut Depot,
281) -> Result<Json<MensajeResponse>, AppError> {
282    let auth_service = depot
283        .obtain::<AuthService>()
284        .map_err(|_| AppError::Internal("AuthService no disponible".into()))?
285        .clone();
286    let auth_repo = depot
287        .obtain::<SeaAuthRepo>()
288        .map_err(|_| AppError::Internal("AuthRepo no disponible".into()))?
289        .clone();
290
291    // Buscar el refresh token en BD y revocarlo
292    let hash = auth_service.hashear_refresh_token(&body.refresh_token);
293    let token_db = auth_repo
294        .buscar_refresh_token_por_hash(&hash)
295        .await?
296        .ok_or(AppError::Unauthorized)?;
297
298    auth_repo.revocar_refresh_token(token_db.id).await?;
299
300    Ok(Json(MensajeResponse {
301        mensaje: "Sesion cerrada".into(),
302    }))
303}

307/// POST /api/v1/auth/firebase — Handshake unico con Firebase ID Token.
308///
309/// La app cliente delega TODO el login en FirebaseAuth (Google, Apple,
310/// email/password, anonimo) y entrega aqui el ID Token JWT firmado por
311/// Google con RS256. El backend lo verifica contra los JWKS de Google,
312/// tipa el `sign_in_provider` con la whitelist `SignInProvider`, y
313/// resuelve un usuario segun el caso:
314///
315///   - **`google.com` / `apple.com` / `password`**: si la identidad ya
316///     existe (mismo `firebase_uid` y mismo provider) la reusa y refresca
317///     su snapshot JSONB. Si no, intenta auto-link por email cuando
318///     `email_verified=true` (asi un usuario legacy con `hash_contrasena`
319///     se enlaza automaticamente; Apple siempre marca el email como
320///     verificado, incluso para relays `@privaterelay.appleid.com`). Si
321///     el email coincide pero NO esta verificado, devuelve 409
322///     (anti-takeover). Si nada coincide, crea un usuario OAuth nuevo.
323///   - **`anonymous`**: crea un usuario sin email/hash y guarda el
324///     `firebase_uid` como `identificador_proveedor`. El JWT propio
325///     emitido lleva `anonimo=true` para que el cliente sepa que falta
326///     completar el registro.
327///   - **Lookup defensivo (`linkWithCredential` en cliente)**: si el
328///     `firebase_uid` ya existe en OTRO provider, reusa el mismo
329///     `id_usuario` y solo anade la nueva fila de identidad. Evita
330///     duplicar usuarios cuando un anonimo se actualiza a Google/Apple/
331///     password conservando su uid.
332///
333/// # Codigos de error
334/// - 400: `id_token` vacio o `sign_in_provider` no soportado todavia
335///   (phone, microsoft.com, github.com, etc.).
336/// - 401: token invalido (firma, expirado, audience, issuer, kid desconocido).
337/// - 409: account linking bloqueado por `email_verified=false`.
338/// - 500: fallo al descargar JWKS de Google o error de BD.
339#[endpoint(tags("Auth"))]
340pub async fn login_firebase(
341    req: &mut Request,
342    body: JsonBody<FirebaseLoginRequest>,
343    depot: &mut Depot,
344) -> Result<Json<AuthResponse>, AppError> {
345    // ── 1. Extraer servicios y repos del Depot ──
346    let auth_service = depot
347        .obtain::<AuthService>()
348        .map_err(|_| AppError::Internal("AuthService no disponible".into()))?
349        .clone();
350    let firebase = depot
351        .obtain::<FirebaseVerifier>()
352        .map_err(|_| AppError::Internal("FirebaseVerifier no disponible".into()))?
353        .clone();
354    let auth_repo = depot
355        .obtain::<SeaAuthRepo>()
356        .map_err(|_| AppError::Internal("AuthRepo no disponible".into()))?
357        .clone();
358    let proveedor_repo = depot
359        .obtain::<SeaProveedorAutenticacionRepo>()
360        .map_err(|_| AppError::Internal("ProveedorAutenticacionRepo no disponible".into()))?
361        .clone();
362
363    // ── 2. Validar input minimo ──
364    if body.id_token.trim().is_empty() {
365        return Err(AppError::BadRequest("id_token es obligatorio".into()));
366    }
367
368    // ── 3. Verificar el ID Token de Firebase (criptografia + claims) ──
369    let claims = firebase.verify(&body.id_token).await?;
370
371    // ── 4. Tipar el provider via whitelist (rechaza phone/microsoft/etc. con 400) ──
372    let provider = claims.provider().map_err(|p| {
373        tracing::warn!(provider = %p, "sign_in_provider de Firebase no soportado todavia");
374        AppError::BadRequest(format!("proveedor `{p}` no soportado todavia"))
375    })?;
376    let anonimo = matches!(provider, SignInProvider::Anonymous);
377
378    // ── 5. Resolver el usuario (upsert + account-linking + lookup defensivo) ──
379    let usuario = upsert_usuario_firebase(&auth_repo, &proveedor_repo, &claims, provider).await?;
380
381    // ── 6. Ultimo login + enriquecer perfil con name/picture (solo no-anonimos) ──
382    auth_repo.actualizar_ultimo_login(usuario.id).await?;
383    if !anonimo {
384        auth_repo
385            .enriquecer_perfil_si_vacio(
386                usuario.id,
387                claims.name.as_deref(),
388                claims.picture.as_deref(),
389            )
390            .await?;
391    }
392
393    // ── 7. Emitir tokens propios (HS256, flag anonimo derivado del provider) ──
394    let access_token = auth_service.generar_access_token_con_flag(usuario.id, anonimo)?;
395    let refresh_raw = auth_service.generar_refresh_token();
396    let refresh_hash = auth_service.hashear_refresh_token(&refresh_raw);
397    let expira = (Utc::now() + Duration::days(30)).fixed_offset();
398    let user_agent = extraer_user_agent(req);
399    auth_repo
400        .guardar_refresh_token(usuario.id, &refresh_hash, user_agent.as_deref(), expira)
401        .await?;
402
403    // ── 8. Recargar el usuario (puede haber cambiado nombre_visible / url_avatar) ──
404    let usuario_final = auth_repo
405        .buscar_usuario_por_id(usuario.id)
406        .await?
407        .unwrap_or(usuario);
408
409    Ok(Json(AuthResponse {
410        access_token,
411        refresh_token: refresh_raw,
412        usuario: construir_usuario_response(
413            &usuario_final,
414            provider.as_str(),
415            anonimo,
416            claims.email_verified,
417        ),
418    }))
419}

421/// Resuelve a que usuario corresponde un login Firebase, creando o
422/// vinculando segun haga falta.
423///
424/// Extraida del handler para mantener el flujo principal legible.
425/// Cubre 5 casos en orden:
426///   1. Identidad EXACTA `(provider, firebase_uid)` ya existente -> reusa.
427///   2. `firebase_uid` existe en OTRO provider (`linkWithCredential`) -> reusa
428///      `id_usuario` y solo anade la nueva fila de identidad.
429///   3. Anonymous nuevo -> crea usuario sin email/hash + identidad.
430///   4. password/google.com/apple.com con email coincidente y verificado ->
431///      auto-link. Apple siempre marca `email_verified=true` (incluso para
432///      relays `@privaterelay.appleid.com`), asi que cae aqui sin friccion.
433///      Si el email NO esta verificado -> 409 (anti-takeover).
434///   5. password/google.com/apple.com sin match -> crea usuario OAuth nuevo
435///      + identidad.
436async fn upsert_usuario_firebase(
437    auth_repo: &SeaAuthRepo,
438    proveedor_repo: &SeaProveedorAutenticacionRepo,
439    claims: &FirebaseClaims,
440    provider: SignInProvider,
441) -> Result<usuario::Model, AppError> {
442    let provider_str = provider.as_str();
443    let datos = construir_datos_proveedor_firebase(claims);
444
445    // 1. Identidad EXACTA (provider, sub) ya existe -> refresca y devuelve.
446    if let Some(prov) = proveedor_repo
447        .buscar_por_proveedor_e_identificador(provider_str, &claims.sub)
448        .await?
449    {
450        proveedor_repo
451            .actualizar_datos(prov.id, claims.email.as_deref(), Some(datos))
452            .await?;
453        return cargar_usuario_obligatorio(auth_repo, prov.id_usuario).await;
454    }
455
456    // 2. Lookup defensivo: el mismo `sub` ya existe en OTRO provider.
457    //    Caso: usuario anonimo que hace `linkWithCredential` en el cliente y
458    //    conserva su firebase_uid pero cambia de provider. Reusamos el
459    //    `id_usuario` y solo anadimos la nueva fila de identidad para no
460    //    crear un usuario duplicado.
461    if let Some(prov) = proveedor_repo
462        .buscar_por_firebase_uid_cualquier_provider(&claims.sub)
463        .await?
464    {
465        tracing::info!(
466            firebase_uid = %claims.sub,
467            provider_existente = %prov.proveedor.as_deref().unwrap_or("(null)"),
468            provider_nuevo = %provider_str,
469            "firebase_uid existente en otro provider, anadiendo nueva identidad al mismo usuario"
470        );
471        crear_proveedor_identidad(proveedor_repo, prov.id_usuario, provider_str, claims, datos)
472            .await?;
473        return cargar_usuario_obligatorio(auth_repo, prov.id_usuario).await;
474    }
475
476    // 3. Anonymous nuevo: crea usuario sin email/hash, fila con identificador=firebase_uid.
477    if matches!(provider, SignInProvider::Anonymous) {
478        let nuevo = auth_repo.crear_usuario_anonimo().await?;
479        crear_proveedor_identidad(proveedor_repo, nuevo.id, provider_str, claims, datos).await?;
480        return Ok(nuevo);
481    }
482
483    // 4. password/google.com/apple.com con email coincidente -> auto-link si
484    //    email_verified=true. Asi un usuario legacy con `hash_contrasena` se
485    //    enlaza automaticamente a su nueva identidad Firebase sin intervencion
486    //    manual. Apple siempre propaga email_verified=true (incluso emails
487    //    relay), por lo que tambien cae aqui sin friccion.
488    if let Some(email) = claims.email.as_deref()
489        && let Some(usuario_existente) = auth_repo.buscar_por_email(email).await?
490    {
491        // Politica: sin verified=true un atacante podria registrar un
492        // proveedor OAuth con un email ajeno y apoderarse de la cuenta local.
493        if claims.email_verified != Some(true) {
494            return Err(AppError::Conflict(
495                "ya existe una cuenta con este email; verifica el email en tu \
496                 proveedor y reintenta para vincular automaticamente"
497                    .into(),
498            ));
499        }
500        crear_proveedor_identidad(
501            proveedor_repo,
502            usuario_existente.id,
503            provider_str,
504            claims,
505            datos,
506        )
507        .await?;
508        return Ok(usuario_existente);
509    }
510
511    // 5. password/google.com/apple.com sin match previo -> usuario OAuth nuevo + identidad.
512    let nuevo = auth_repo
513        .crear_usuario_oauth(
514            claims.email.as_deref(),
515            claims.name.as_deref(),
516            claims.picture.as_deref(),
517        )
518        .await?;
519    crear_proveedor_identidad(proveedor_repo, nuevo.id, provider_str, claims, datos).await?;
520    Ok(nuevo)
521}

523/// Inserta una fila en `proveedores_autenticacion` que representa la
524/// identidad Firebase de un usuario en un provider concreto.
525///
526/// `provider_str` es el literal `sign_in_provider` (ya validado contra
527/// la whitelist tipada `SignInProvider`); ej: `"google.com"`, `"password"`,
528/// `"anonymous"`. Se guarda en la columna `proveedor`, mientras que
529/// `claims.sub` (el `firebase_uid`) va a `identificador_proveedor`.
530async fn crear_proveedor_identidad(
531    proveedor_repo: &SeaProveedorAutenticacionRepo,
532    id_usuario: Uuid,
533    provider_str: &str,
534    claims: &FirebaseClaims,
535    datos: serde_json::Value,
536) -> Result<proveedor_autenticacion::Model, AppError> {
537    proveedor_repo
538        .crear(
539            id_usuario,
540            provider_str,
541            Some(&claims.sub),
542            claims.email.as_deref(),
543            Some(datos),
544        )
545        .await
546}
