60impl ReporteRepo for SeaReporteRepo {
61    async fn crear_reporte(
62        &self,
63        id_reportero: Uuid,
64        datos: CrearReporteDto,
65    ) -> Result<reporte::Model, AppError> {
66        // Verificar que no haya un reporte duplicado del mismo usuario
67        let existente = reporte::Entity::find()
68            .filter(reporte::Column::TipoContenido.eq(Some(datos.tipo_contenido.clone())))
69            .filter(reporte::Column::IdContenido.eq(Some(datos.id_contenido)))
70            .filter(reporte::Column::IdReportero.eq(id_reportero))
71            .one(&self.db)
72            .await
73            .map_err(AppError::from_db)?;
74
75        if existente.is_some() {
76            return Err(AppError::Conflict("Ya has reportado este contenido".into()));
77        }
78
79        let nuevo = reporte::ActiveModel {
80            id: Set(Uuid::new_v4()),
81            tipo_contenido: Set(Some(datos.tipo_contenido)),
82            id_contenido: Set(Some(datos.id_contenido)),
83            id_reportero: Set(id_reportero),
84            motivo: Set(Some(datos.motivo)),
85            detalle_motivo: Set(datos.detalle_motivo),
86            ..Default::default()
87        };
88
89        nuevo.insert(&self.db).await.map_err(AppError::from_db)
90    }
91
92    async fn listar_reportes_pendientes(&self) -> Result<Vec<reporte::Model>, AppError> {
93        reporte::Entity::find()
94            .filter(reporte::Column::Estado.eq(Some(EstadoReporte::Pendiente)))
95            .all(&self.db)
96            .await
97            .map_err(AppError::from_db)
98    }
99
100    async fn obtener_reporte(&self, id: Uuid) -> Result<reporte::Model, AppError> {
101        reporte::Entity::find_by_id(id)
102            .one(&self.db)
103            .await
104            .map_err(AppError::from_db)?
105            .ok_or_else(|| AppError::NotFound(format!("Reporte con id {id}")))
106    }
107
108    async fn eliminar_reporte(&self, id: Uuid) -> Result<(), AppError> {
109        let result = reporte::Entity::delete_by_id(id)
110            .exec(&self.db)
111            .await
112            .map_err(AppError::from_db)?;
113
114        if result.rows_affected == 0 {
115            return Err(AppError::NotFound(format!("Reporte con id {id}")));
116        }
117        Ok(())
118    }
119
120    async fn procesar_reporte(
121        &self,
122        id: Uuid,
123        id_procesador: Uuid,
124        aceptar: bool,
125    ) -> Result<reporte::Model, AppError> {
126        let reporte = reporte::Entity::find_by_id(id)
127            .one(&self.db)
128            .await
129            .map_err(AppError::from_db)?
130            .ok_or_else(|| AppError::NotFound(format!("Reporte con id {id}")))?;
131
132        let mut active: reporte::ActiveModel = reporte.into();
133        active.estado = Set(Some(if aceptar {
134            EstadoReporte::Aceptado
135        } else {
136            EstadoReporte::Rechazado
137        }));
138        active.id_procesador = Set(Some(id_procesador));
139        active.procesado_en = Set(Some(Utc::now().fixed_offset()));
140
141        active.update(&self.db).await.map_err(AppError::from_db)
142    }
143}
