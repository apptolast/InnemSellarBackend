1// src/services/firebase_verifier.rs

3// Verificador de Firebase ID Tokens — fase: login con Google y anonimo.
4//
5// El cliente Flutter completa el login con FirebaseAuth en su lado y obtiene
6// un ID Token JWT firmado con RS256 por Google. El backend NO confia en ese
7// token sin verificar: descarga la clave publica del JWKS de Google, valida
8// la firma y los claims (issuer, audience, expiracion) y solo entonces
9// considera valido al usuario.
10//
11// NO accede a la BD — eso lo hace el repositorio en `proveedor_autenticacion_repo`.
12// NO maneja HTTP de la API — eso lo hace el handler en `handlers/auth.rs`.
13

25/// URL publica del JWKS de Google que firma los Firebase ID Tokens.
26///
27/// # Por que esta URL y no `/x509/...`
28/// Google publica las claves publicas en dos formatos:
29///   - JWKS (JSON Web Key Set) — formato JSON con `kty`, `n`, `e`, `kid`. ESTE.
30///   - x509 — certificados PEM en JSON.
31///
32/// La URL JWKS es la canonica y la que mejor encaja con `jsonwebtoken` 9.
33/// La eligio el equipo de Firebase como fuente oficial.
34const JWKS_URL_DEFAULT: &str =
35    "https://www.googleapis.com/service_accounts/v1/jwk/securetoken@system.gserviceaccount.com";
36
37/// TTL del cache de JWKS si la respuesta no incluye `Cache-Control: max-age=...`.
38/// 1 hora es conservador: las claves de Google rotan cada varias horas.
39const FALLBACK_TTL: Duration = Duration::from_secs(60 * 60);
40
41/// Tolerancia de reloj para `exp`, `iat`, `auth_time`. 60 segundos es el
42/// estandar de la industria: corrige drift NTP sin abrir ventana de replay.
43const CLOCK_SKEW_SECS: u64 = 60;
44
45/// Timeout duro para la descarga del JWKS. Si Google tarda mas, fallamos
46/// rapido en vez de bloquear handlers eternamente.
47const JWKS_FETCH_TIMEOUT: Duration = Duration::from_secs(10);
48

51/// Datos del proveedor concreto que firmo el token (claim `firebase`).
52///
53/// Firebase Auth usa este bloque para distinguir entre los distintos
54/// providers (google.com, password, anonymous, apple.com...) que pueden
55/// haber autenticado al usuario.
56///
57/// # Por que `#[serde(default)]` en `identities`
58/// Para tokens de usuarios anonimos, `identities` esta presente como objeto
59/// vacio `{}`. Para tokens de Google viene rellenado con
60/// `{ "google.com": ["1234567890"], "email": ["alice@gmail.com"] }`.
61/// `default` evita errores si Firebase omite el campo en algun futuro.
62#[derive(Debug, Clone, Deserialize, Serialize)]
63pub struct FirebaseProviderInfo {
64    /// Identifica el proveedor: `"google.com"`, `"password"`, `"anonymous"`,
65    /// `"apple.com"`, etc.
66    pub sign_in_provider: String,
67
68    /// Identidades por proveedor, ej: `{ "google.com": ["uid1"], "email": [...] }`.
69    /// Util para Account Linking (un usuario con multiples identidades).
70    #[serde(default)]
71    pub identities: HashMap<String, serde_json::Value>,
72
73    /// Solo presente si Firebase Auth se uso con multitenancy.
74    /// En InemSellar no se usa, asi que normalmente sera `None`.
75    #[serde(default)]
76    pub tenant: Option<String>,
77}

79/// Claims completos de un Firebase ID Token tras la validacion criptografica.
80///
81/// # Por que tantos `Option<...>`
82/// Solo los claims estandar JWT (iss, aud, sub, iat, exp, auth_time) estan
83/// SIEMPRE presentes — los validamos como `Required` en `Validation`.
84/// Los demas dependen del proveedor: un usuario anonimo no tiene `email`,
85/// un usuario que se acaba de registrar no tiene `picture`, etc.
86/// `Option` modela esto correctamente sin asumir nada.
87#[derive(Debug, Clone, Deserialize, Serialize)]
88pub struct FirebaseClaims {
89    /// Issuer — debe ser `https://securetoken.google.com/<project_id>`.
90    pub iss: String,
91    /// Audience — debe ser `<project_id>` exactamente.
92    pub aud: String,
93    /// Subject — el `firebase_uid` del usuario. Identificador estable.
94    pub sub: String,
95    /// Issued At (Unix epoch en segundos).
96    pub iat: i64,
97    /// Expiration (Unix epoch en segundos).
98    pub exp: i64,
99    /// Cuando el usuario se autentico (puede ser anterior a `iat` si refresco).
100    pub auth_time: i64,
101
102    /// Email del usuario, si el proveedor lo proporciona.
103    /// Vacio para anonimos y para usuarios de phone-only.
104    #[serde(default)]
105    pub email: Option<String>,
106    /// `true` si el proveedor (Google, etc.) ha verificado el email.
107    #[serde(default)]
108    pub email_verified: Option<bool>,
109    /// Nombre completo segun el proveedor.
110    #[serde(default)]
111    pub name: Option<String>,
112    /// URL del avatar segun el proveedor.
113    #[serde(default)]
114    pub picture: Option<String>,
115
116    /// Bloque con el provider y las identidades.
117    pub firebase: FirebaseProviderInfo,
118}

120impl FirebaseClaims {
121    /// `true` si el token corresponde a un usuario anonimo de Firebase.
122    pub fn is_anonymous(&self) -> bool {
123        self.firebase.sign_in_provider == "anonymous"
124    }
125
126    /// `true` si el token corresponde a un login con Google.
127    pub fn is_google(&self) -> bool {
128        self.firebase.sign_in_provider == "google.com"
129    }
130}

132// ─── Cache de JWKS ────────────────────────────────────────────────────────

134/// Estado interno del cache de claves publicas.
135struct JwksCache {
136    /// `None` mientras no se haya descargado nunca; `Some(set)` despues.
137    set: Option<JwkSet>,
138    /// Momento a partir del cual el cache esta caducado. Derivado de
139    /// `Cache-Control: max-age` de la respuesta de Google, con `FALLBACK_TTL`
140    /// si la cabecera no se puede parsear.
141    expires_at: Instant,
142}

144// ─── FirebaseVerifier ─────────────────────────────────────────────────────
145
146/// Verificador de Firebase ID Tokens.
147///
148/// # Por que `Clone`
149/// Salvo necesita que los servicios inyectados en `Depot` sean `Clone`.
150/// Internamente todo es `Arc<...>`, asi que clonar es barato (un AtomicUsize++).
151///
152/// # Por que `RwLock` y no `Mutex` para el cache
153/// El 99% de las llamadas son LECTURAS del cache (verificar tokens). Solo
154/// cuando el cache caduca o no encontramos un `kid` necesitamos ESCRIBIR
155/// (refrescar). `RwLock` permite multiples lectores concurrentes y solo
156/// bloquea cuando hay un escritor — exactamente nuestro patron de acceso.
157///
158/// # Por que un `Mutex` adicional para `refresh_lock`
159/// Sin el, varias peticiones simultaneas que vean el cache caducado
160/// dispararian fetches en paralelo al JWKS (thundering herd). Con
161/// `refresh_lock`, solo una hace el fetch; las demas esperan y reusan
162/// el cache recien actualizado (double-checked locking).
163#[derive(Clone)]
164pub struct FirebaseVerifier {
165    /// `project_id` de Firebase. Validado contra el claim `aud`.
166    project_id: Arc<String>,
167    /// Issuer esperado: `https://securetoken.google.com/<project_id>`.
168    issuer: Arc<String>,
169    /// URL del JWKS. Configurable para tests con `new_with_url`.
170    jwks_url: Arc<String>,
171    /// Cliente HTTP reutilizable. `reqwest::Client` ya envuelve un Arc internamente.
172    http: reqwest::Client,
173    /// Cache de claves publicas con TTL.
174    cache: Arc<RwLock<JwksCache>>,
175    /// Mutex para serializar refrescos del cache (evita thundering herd).
176    refresh_lock: Arc<Mutex<()>>,
177}

179impl FirebaseVerifier {
180    /// Crea un verificador apuntando al JWKS oficial de Google.
181    ///
182    /// # Panics
183    /// Si reqwest no puede construir su cliente (extremadamente raro, suele
184    /// indicar un sistema sin TLS roots). Aceptable porque ocurre solo al
185    /// arrancar el servidor — fallo rapido es preferible a degradacion.
186    pub fn new(project_id: String) -> Self {
187        Self::new_with_url(project_id, JWKS_URL_DEFAULT.to_string())
188    }
189
190    /// Constructor parametrizable. Usado en tests con `wiremock` para
191    /// servir un JWKS mockeado en lugar del de Google.
192    pub fn new_with_url(project_id: String, jwks_url: String) -> Self {
193        let issuer = format!("https://securetoken.google.com/{project_id}");
194        let http = reqwest::Client::builder()
195            .timeout(JWKS_FETCH_TIMEOUT)
196            .build()
197            .expect("reqwest::Client::builder() fallo al construir cliente HTTP");
198
199        Self {
200            project_id: Arc::new(project_id),
201            issuer: Arc::new(issuer),
202            jwks_url: Arc::new(jwks_url),
203            http,
204            cache: Arc::new(RwLock::new(JwksCache {
205                set: None,
206                // `Instant::now()` ya esta caducado: el primer `verify` forzara
207                // un fetch del JWKS. Es el comportamiento deseado.
208                expires_at: Instant::now(),
209            })),
210            refresh_lock: Arc::new(Mutex::new(())),
211        }
212    }
213
214    /// Verifica un Firebase ID Token y devuelve los claims si es valido.
215    ///
216    /// # Por que `AppError::Unauthorized` para todos los fallos de validacion
217    /// Desde el punto de vista del cliente, cualquier fallo de verificacion
218    /// (firma, expiracion, audience, issuer, kid desconocido, etc.) significa
219    /// lo mismo: "tu token no sirve". Distinguir entre subcategorias daria
220    /// pistas a un atacante sobre QUE parte de su token forjado fallo.
221    /// Por eso todos colapsan a 401 — y se loguea el motivo real para el operador.
222    ///
223    /// # Por que algunos errores son `Internal`
224    /// Si falla la red al descargar el JWKS, no es culpa del cliente.
225    /// Devolver 500 indica al cliente que reintente; devolver 401 le diria
226    /// que su token es invalido (mentira).
227    pub async fn verify(&self, id_token: &str) -> Result<FirebaseClaims, AppError> {
228        // 1. Decodificar la cabecera SIN validar firma para sacar el `kid`.
229        //    Necesitamos el `kid` para saber que clave publica buscar.
230        let header = decode_header(id_token).map_err(|e| {
231            tracing::warn!(error = %e, "Header de Firebase ID Token invalido");
232            AppError::Unauthorized
233        })?;
234        let kid = header.kid.ok_or_else(|| {
235            tracing::warn!("Firebase ID Token sin claim `kid` en el header");
236            AppError::Unauthorized
237        })?;
238
239        // 2. Buscar la JWK en cache. Si no esta y el cache esta poblado,
240        //    forzar UN unico refresh y reintentar (las claves rotaron).
241        let jwk = match self.find_jwk(&kid).await? {
242            Some(j) => j,
243            None => {
244                tracing::info!(kid = %kid, "kid no en cache JWKS, forzando refresh");
245                self.refresh_jwks().await?;
246                self.find_jwk(&kid).await?.ok_or_else(|| {
247                    tracing::warn!(kid = %kid, "kid no encontrado en JWKS de Google tras refresh");
248                    AppError::Unauthorized
249                })?
250            }
251        };
252
253        // 3. Construir DecodingKey desde la JWK.
254        //    Por seguridad solo aceptamos RSA — Firebase usa RS256, nada mas.
255        let decoding_key = match &jwk.algorithm {
256            AlgorithmParameters::RSA(rsa) => DecodingKey::from_rsa_components(&rsa.n, &rsa.e)
257                .map_err(|e| AppError::Internal(format!("JWK RSA malformada: {e}")))?,
258            other => {
259                tracing::warn!(?other, "Tipo de JWK no soportado, esperaba RSA");
260                return Err(AppError::Unauthorized);
261            }
262        };
263
264        // 4. Configurar la validacion estricta:
265        //    - Algoritmo PINNED a RS256 (no HS256, no none — protege contra alg confusion).
266        //    - audience = project_id.
267        //    - issuer = securetoken.google.com/<project_id>.
268        //    - claims minimos requeridos.
269        //    - leeway de 60s para clock skew.
270        let mut validation = Validation::new(Algorithm::RS256);
271        validation.set_audience(&[self.project_id.as_str()]);
272        validation.set_issuer(&[self.issuer.as_str()]);
273        validation.set_required_spec_claims(&["exp", "iat", "aud", "iss", "sub"]);
274        validation.leeway = CLOCK_SKEW_SECS;
275
276        // 5. Decodificar y validar firma + claims.
277        let data = decode::<FirebaseClaims>(id_token, &decoding_key, &validation).map_err(|e| {
278            tracing::warn!(error = %e, "Validacion Firebase ID Token fallo");
279            AppError::Unauthorized
280        })?;
281        let claims = data.claims;
282
283        // 6. Comprobaciones manuales que `jsonwebtoken` no hace:
284        //    - `sub` no vacio (es el firebase_uid).
285        //    - `auth_time` no en el futuro (con tolerancia de reloj).
286        if claims.sub.trim().is_empty() {
287            tracing::warn!("Firebase ID Token con `sub` vacio");
288            return Err(AppError::Unauthorized);
289        }
290        let now = chrono::Utc::now().timestamp();
291        if claims.auth_time > now + CLOCK_SKEW_SECS as i64 {
292            tracing::warn!(auth_time = claims.auth_time, now, "auth_time en el futuro");
293            return Err(AppError::Unauthorized);
294        }
295
296        Ok(claims)
297    }

299    /// Busca una `Jwk` por `kid` en el cache, refrescando si esta caducado.
300    async fn find_jwk(&self, kid: &str) -> Result<Option<Jwk>, AppError> {
301        if self.is_expired().await {
302            self.refresh_jwks().await?;
303        }
304        let cache = self.cache.read().await;
305        Ok(cache.set.as_ref().and_then(|set| set.find(kid).cloned()))
306    }
307
308    /// `true` si el cache esta vacio o ha caducado.
309    async fn is_expired(&self) -> bool {
310        let cache = self.cache.read().await;
311        cache.set.is_none() || Instant::now() >= cache.expires_at
312    }
313
314    /// Refresca el cache de JWKS desde Google.
315    ///
316    /// # Double-checked locking
317    /// Si llegan 50 requests simultaneas con el cache caducado, sin esta
318    /// proteccion las 50 dispararian fetches paralelos al JWKS (thundering
319    /// herd). Con el `refresh_lock`, solo la primera entra; las demas esperan
320    /// el guard, y al entrar comprueban de nuevo `is_expired()`: si el primero
321    /// ya refresco, no vuelven a fetchear.
322    async fn refresh_jwks(&self) -> Result<(), AppError> {
323        let _guard = self.refresh_lock.lock().await;
324
325        // Double-check: el cache puede haberse refrescado mientras esperabamos.
326        if !self.is_expired().await {
327            return Ok(());
328        }
329
330        let resp = self
331            .http
332            .get(self.jwks_url.as_str())
333            .send()
334            .await
335            .map_err(|e| AppError::Internal(format!("error de red al descargar JWKS: {e}")))?
336            .error_for_status()
337            .map_err(|e| AppError::Internal(format!("HTTP no-OK al descargar JWKS: {e}")))?;
338
339        // Respetar `Cache-Control: max-age=N` de Google.
340        // Si la cabecera no se puede parsear, usar FALLBACK_TTL.
341        let ttl = resp
342            .headers()
343            .get(reqwest::header::CACHE_CONTROL)
344            .and_then(|v| v.to_str().ok())
345            .and_then(parse_max_age)
346            .map(Duration::from_secs)
347            .unwrap_or(FALLBACK_TTL);
348
349        let body = resp
350            .bytes()
351            .await
352            .map_err(|e| AppError::Internal(format!("error leyendo cuerpo JWKS: {e}")))?;
353        let set: JwkSet = serde_json::from_slice(&body)
354            .map_err(|e| AppError::Internal(format!("JWKS malformado: {e}")))?;
355
356        let mut cache = self.cache.write().await;
357        cache.set = Some(set);
358        cache.expires_at = Instant::now() + ttl;
359        tracing::info!(ttl_secs = ttl.as_secs(), "JWKS de Firebase refrescado");
360        Ok(())
361    }

364/// Parsea el valor `max-age=N` de un header `Cache-Control`.
365///
366/// Acepta formatos como:
367///   - `public, max-age=21600`
368///   - `max-age=300, must-revalidate`
369///   - `max-age=0` (devuelve 0)
370///
371/// Devuelve `None` si no encuentra `max-age=` o si el numero no es valido.
372fn parse_max_age(header: &str) -> Option<u64> {
373    header
374        .split(',')
375        .map(str::trim)
376        .find_map(|p| p.strip_prefix("max-age=").and_then(|n| n.parse().ok()))
377}
