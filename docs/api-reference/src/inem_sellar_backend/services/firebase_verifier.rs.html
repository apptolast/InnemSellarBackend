1// src/services/firebase_verifier.rs

3// Verificador de Firebase ID Tokens — fase: login con Google y anonimo.
4//
5// El cliente Flutter completa el login con FirebaseAuth en su lado y obtiene
6// un ID Token JWT firmado con RS256 por Google. El backend NO confia en ese
7// token sin verificar: descarga la clave publica del JWKS de Google, valida
8// la firma y los claims (issuer, audience, expiracion) y solo entonces
9// considera valido al usuario.
10//
11// NO accede a la BD — eso lo hace el repositorio en `proveedor_autenticacion_repo`.
12// NO maneja HTTP de la API — eso lo hace el handler en `handlers/auth.rs`.
13

25/// URL publica del JWKS de Google que firma los Firebase ID Tokens.
26///
27/// # Por que esta URL y no `/x509/...`
28/// Google publica las claves publicas en dos formatos:
29///   - JWKS (JSON Web Key Set) — formato JSON con `kty`, `n`, `e`, `kid`. ESTE.
30///   - x509 — certificados PEM en JSON.
31///
32/// La URL JWKS es la canonica y la que mejor encaja con `jsonwebtoken` 9.
33/// La eligio el equipo de Firebase como fuente oficial.
34const JWKS_URL_DEFAULT: &str =
35    "https://www.googleapis.com/service_accounts/v1/jwk/securetoken@system.gserviceaccount.com";
36
37/// TTL del cache de JWKS si la respuesta no incluye `Cache-Control: max-age=...`.
38/// 1 hora es conservador: las claves de Google rotan cada varias horas.
39const FALLBACK_TTL: Duration = Duration::from_secs(60 * 60);
40
41/// Tolerancia de reloj para `exp`, `iat`, `auth_time`. 60 segundos es el
42/// estandar de la industria: corrige drift NTP sin abrir ventana de replay.
43const CLOCK_SKEW_SECS: u64 = 60;
44
45/// Timeout duro para la descarga del JWKS. Si Google tarda mas, fallamos
46/// rapido en vez de bloquear handlers eternamente.
47const JWKS_FETCH_TIMEOUT: Duration = Duration::from_secs(10);

49// ─── Provider whitelist tipada ────────────────────────────────────────────

51/// Conjunto de Firebase `sign_in_provider` que el backend acepta.
52///
53/// # Por que un enum en vez de comparaciones de string sueltas
54/// El `match` exhaustivo en el handler dejara de COMPILAR el dia que se
55/// anada `Apple` y se olvide cubrir esa rama. Centraliza los literales
56/// en un solo sitio y evita que un provider nuevo pase silenciosamente
57/// por el camino equivocado.
58///
59/// # Por que `Copy`
60/// El enum no tiene datos asociados (3 variantes unitarias), asi que
61/// copiarlo es trivial y permite pasarlo por valor sin ceremonia
62/// (`fn(provider: SignInProvider)` en vez de `&SignInProvider`).
63#[derive(Debug, Clone, Copy, PartialEq, Eq)]
64pub enum SignInProvider {
65    /// Firebase OAuth con Google. `sign_in_provider == "google.com"`.
66    GoogleCom,
67    /// Firebase OAuth con Apple. `sign_in_provider == "apple.com"`.
68    ///
69    /// En iOS via AuthenticationServices nativo (Sign in with Apple del
70    /// sistema), en Android via Firebase OAuth web (Chrome Custom Tab
71    /// abriendo `appleid.apple.com`). En ambos casos Firebase emite el
72    /// mismo ID Token RS256 que validamos contra el JWKS de Google.
73    AppleCom,
74    /// Firebase Email/Password. `sign_in_provider == "password"`.
75    Password,
76    /// Firebase Anonymous Auth. `sign_in_provider == "anonymous"`.
77    Anonymous,
78}
79
80impl SignInProvider {
81    /// Devuelve el literal Firebase que se guarda en
82    /// `proveedores_autenticacion.proveedor` y se serializa en
83    /// `usuario.proveedor` de la respuesta JSON al cliente.
84    ///
85    /// El round-trip `try_from(p.as_str()) == Ok(p)` se verifica en tests.
86    pub fn as_str(&self) -> &'static str {
87        match self {
88            Self::GoogleCom => "google.com",
89            Self::AppleCom => "apple.com",
90            Self::Password => "password",
91            Self::Anonymous => "anonymous",
92        }
93    }
94}
95
96impl TryFrom<&str> for SignInProvider {
97    type Error = String;
98
99    /// Convierte el `sign_in_provider` recibido de Firebase a un
100    /// `SignInProvider` tipado.
101    ///
102    /// # Por que `Err(String)` y no un enum de error
103    /// El caller (handler `login_firebase`) usa el literal desconocido
104    /// para construir un mensaje 400 informativo del estilo
105    /// `"proveedor `phone` no soportado todavia"`. Devolver el string
106    /// original ahorra una capa de mapeo y mantiene el caller delgado.
107    fn try_from(value: &str) -> Result<Self, Self::Error> {
108        match value {
109            "google.com" => Ok(Self::GoogleCom),
110            "apple.com" => Ok(Self::AppleCom),
111            "password" => Ok(Self::Password),
112            "anonymous" => Ok(Self::Anonymous),
113            other => Err(other.to_string()),
114        }
115    }
116}

118// ─── Claims del Firebase ID Token ─────────────────────────────────────────
119
120/// Datos del proveedor concreto que firmo el token (claim `firebase`).
121///
122/// Firebase Auth usa este bloque para distinguir entre los distintos
123/// providers (google.com, password, anonymous, apple.com...) que pueden
124/// haber autenticado al usuario.
125///
126/// # Por que `#[serde(default)]` en `identities`
127/// Para tokens de usuarios anonimos, `identities` esta presente como objeto
128/// vacio `{}`. Para tokens de Google viene rellenado con
129/// `{ "google.com": ["1234567890"], "email": ["alice@gmail.com"] }`.
130/// `default` evita errores si Firebase omite el campo en algun futuro.
131#[derive(Debug, Clone, Deserialize, Serialize)]
132pub struct FirebaseProviderInfo {
133    /// Identifica el proveedor: `"google.com"`, `"password"`, `"anonymous"`,
134    /// `"apple.com"`, etc.
135    pub sign_in_provider: String,
136
137    /// Identidades por proveedor, ej: `{ "google.com": ["uid1"], "email": [...] }`.
138    /// Util para Account Linking (un usuario con multiples identidades).
139    #[serde(default)]
140    pub identities: HashMap<String, serde_json::Value>,
141
142    /// Solo presente si Firebase Auth se uso con multitenancy.
143    /// En InemSellar no se usa, asi que normalmente sera `None`.
144    #[serde(default)]
145    pub tenant: Option<String>,
146}
147
148/// Claims completos de un Firebase ID Token tras la validacion criptografica.
149///
150/// # Por que tantos `Option<...>`
151/// Solo los claims estandar JWT (iss, aud, sub, iat, exp, auth_time) estan
152/// SIEMPRE presentes — los validamos como `Required` en `Validation`.
153/// Los demas dependen del proveedor: un usuario anonimo no tiene `email`,
154/// un usuario que se acaba de registrar no tiene `picture`, etc.
155/// `Option` modela esto correctamente sin asumir nada.
156#[derive(Debug, Clone, Deserialize, Serialize)]
157pub struct FirebaseClaims {
158    /// Issuer — debe ser `https://securetoken.google.com/<project_id>`.
159    pub iss: String,
160    /// Audience — debe ser `<project_id>` exactamente.
161    pub aud: String,
162    /// Subject — el `firebase_uid` del usuario. Identificador estable.
163    pub sub: String,
164    /// Issued At (Unix epoch en segundos).
165    pub iat: i64,
166    /// Expiration (Unix epoch en segundos).
167    pub exp: i64,
168    /// Cuando el usuario se autentico (puede ser anterior a `iat` si refresco).
169    pub auth_time: i64,
170
171    /// Email del usuario, si el proveedor lo proporciona.
172    /// Vacio para anonimos y para usuarios de phone-only.
173    #[serde(default)]
174    pub email: Option<String>,
175    /// `true` si el proveedor (Google, etc.) ha verificado el email.
176    #[serde(default)]
177    pub email_verified: Option<bool>,
178    /// Nombre completo segun el proveedor.
179    #[serde(default)]
180    pub name: Option<String>,
181    /// URL del avatar segun el proveedor.
182    #[serde(default)]
183    pub picture: Option<String>,
184
185    /// Bloque con el provider y las identidades.
186    pub firebase: FirebaseProviderInfo,
187}
188
189impl FirebaseClaims {
190    /// Convierte el campo `sign_in_provider` del claim `firebase` a un
191    /// `SignInProvider` tipado.
192    ///
193    /// Devuelve `Err(literal)` si el provider no esta en la whitelist
194    /// (Apple, phone, etc., todavia no soportados). El caller
195    /// (`login_firebase`) usa ese literal para construir un 400 con un
196    /// mensaje accionable para el cliente.
197    pub fn provider(&self) -> Result<SignInProvider, String> {
198        SignInProvider::try_from(self.firebase.sign_in_provider.as_str())
199    }
200}

204/// Estado interno del cache de claves publicas.
205struct JwksCache {
206    /// `None` mientras no se haya descargado nunca; `Some(set)` despues.
207    set: Option<JwkSet>,
208    /// Momento a partir del cual el cache esta caducado. Derivado de
209    /// `Cache-Control: max-age` de la respuesta de Google, con `FALLBACK_TTL`
210    /// si la cabecera no se puede parsear.
211    expires_at: Instant,
212}

214// ─── FirebaseVerifier ─────────────────────────────────────────────────────
215
216/// Verificador de Firebase ID Tokens.
217///
218/// # Por que `Clone`
219/// Salvo necesita que los servicios inyectados en `Depot` sean `Clone`.
220/// Internamente todo es `Arc<...>`, asi que clonar es barato (un AtomicUsize++).
221///
222/// # Por que `RwLock` y no `Mutex` para el cache
223/// El 99% de las llamadas son LECTURAS del cache (verificar tokens). Solo
224/// cuando el cache caduca o no encontramos un `kid` necesitamos ESCRIBIR
225/// (refrescar). `RwLock` permite multiples lectores concurrentes y solo
226/// bloquea cuando hay un escritor — exactamente nuestro patron de acceso.
227///
228/// # Por que un `Mutex` adicional para `refresh_lock`
229/// Sin el, varias peticiones simultaneas que vean el cache caducado
230/// dispararian fetches en paralelo al JWKS (thundering herd). Con
231/// `refresh_lock`, solo una hace el fetch; las demas esperan y reusan
232/// el cache recien actualizado (double-checked locking).
233#[derive(Clone)]
234pub struct FirebaseVerifier {
235    /// `project_id` de Firebase. Validado contra el claim `aud`.
236    project_id: Arc<String>,
237    /// Issuer esperado: `https://securetoken.google.com/<project_id>`.
238    issuer: Arc<String>,
239    /// URL del JWKS. Configurable para tests con `new_with_url`.
240    jwks_url: Arc<String>,
241    /// Cliente HTTP reutilizable. `reqwest::Client` ya envuelve un Arc internamente.
242    http: reqwest::Client,
243    /// Cache de claves publicas con TTL.
244    cache: Arc<RwLock<JwksCache>>,
245    /// Mutex para serializar refrescos del cache (evita thundering herd).
246    refresh_lock: Arc<Mutex<()>>,
247}
248
249impl FirebaseVerifier {
250    /// Crea un verificador apuntando al JWKS oficial de Google.
251    ///
252    /// # Panics
253    /// Si reqwest no puede construir su cliente (extremadamente raro, suele
254    /// indicar un sistema sin TLS roots). Aceptable porque ocurre solo al
255    /// arrancar el servidor — fallo rapido es preferible a degradacion.
256    pub fn new(project_id: String) -> Self {
257        Self::new_with_url(project_id, JWKS_URL_DEFAULT.to_string())
258    }
259
260    /// Constructor parametrizable. Usado en tests con `wiremock` para
261    /// servir un JWKS mockeado en lugar del de Google.
262    pub fn new_with_url(project_id: String, jwks_url: String) -> Self {
263        let issuer = format!("https://securetoken.google.com/{project_id}");
264        let http = reqwest::Client::builder()
265            .timeout(JWKS_FETCH_TIMEOUT)
266            .build()
267            .expect("reqwest::Client::builder() fallo al construir cliente HTTP");
268
269        Self {
270            project_id: Arc::new(project_id),
271            issuer: Arc::new(issuer),
272            jwks_url: Arc::new(jwks_url),
273            http,
274            cache: Arc::new(RwLock::new(JwksCache {
275                set: None,
276                // `Instant::now()` ya esta caducado: el primer `verify` forzara
277                // un fetch del JWKS. Es el comportamiento deseado.
278                expires_at: Instant::now(),
279            })),
280            refresh_lock: Arc::new(Mutex::new(())),
281        }
282    }
283
284    /// Verifica un Firebase ID Token y devuelve los claims si es valido.
285    ///
286    /// # Por que `AppError::Unauthorized` para todos los fallos de validacion
287    /// Desde el punto de vista del cliente, cualquier fallo de verificacion
288    /// (firma, expiracion, audience, issuer, kid desconocido, etc.) significa
289    /// lo mismo: "tu token no sirve". Distinguir entre subcategorias daria
290    /// pistas a un atacante sobre QUE parte de su token forjado fallo.
291    /// Por eso todos colapsan a 401 — y se loguea el motivo real para el operador.
292    ///
293    /// # Por que algunos errores son `Internal`
294    /// Si falla la red al descargar el JWKS, no es culpa del cliente.
295    /// Devolver 500 indica al cliente que reintente; devolver 401 le diria
296    /// que su token es invalido (mentira).
297    pub async fn verify(&self, id_token: &str) -> Result<FirebaseClaims, AppError> {
298        // 1. Decodificar la cabecera SIN validar firma para sacar el `kid`.
299        //    Necesitamos el `kid` para saber que clave publica buscar.
300        let header = decode_header(id_token).map_err(|e| {
301            tracing::warn!(error = %e, "Header de Firebase ID Token invalido");
302            AppError::Unauthorized
303        })?;
304        let kid = header.kid.ok_or_else(|| {
305            tracing::warn!("Firebase ID Token sin claim `kid` en el header");
306            AppError::Unauthorized
307        })?;
308
309        // 2. Buscar la JWK en cache. Si no esta y el cache esta poblado,
310        //    forzar UN unico refresh y reintentar (las claves rotaron).
311        let jwk = match self.find_jwk(&kid).await? {
312            Some(j) => j,
313            None => {
314                tracing::info!(kid = %kid, "kid no en cache JWKS, forzando refresh");
315                self.refresh_jwks().await?;
316                self.find_jwk(&kid).await?.ok_or_else(|| {
317                    tracing::warn!(kid = %kid, "kid no encontrado en JWKS de Google tras refresh");
318                    AppError::Unauthorized
319                })?
320            }
321        };
322
323        // 3. Construir DecodingKey desde la JWK.
324        //    Por seguridad solo aceptamos RSA — Firebase usa RS256, nada mas.
325        let decoding_key = match &jwk.algorithm {
326            AlgorithmParameters::RSA(rsa) => DecodingKey::from_rsa_components(&rsa.n, &rsa.e)
327                .map_err(|e| AppError::Internal(format!("JWK RSA malformada: {e}")))?,
328            other => {
329                tracing::warn!(?other, "Tipo de JWK no soportado, esperaba RSA");
330                return Err(AppError::Unauthorized);
331            }
332        };
333
334        // 4. Configurar la validacion estricta:
335        //    - Algoritmo PINNED a RS256 (no HS256, no none — protege contra alg confusion).
336        //    - audience = project_id.
337        //    - issuer = securetoken.google.com/<project_id>.
338        //    - claims minimos requeridos.
339        //    - leeway de 60s para clock skew.
340        let mut validation = Validation::new(Algorithm::RS256);
341        validation.set_audience(&[self.project_id.as_str()]);
342        validation.set_issuer(&[self.issuer.as_str()]);
343        validation.set_required_spec_claims(&["exp", "iat", "aud", "iss", "sub"]);
344        validation.leeway = CLOCK_SKEW_SECS;
345
346        // 5. Decodificar y validar firma + claims.
347        let data = decode::<FirebaseClaims>(id_token, &decoding_key, &validation).map_err(|e| {
348            tracing::warn!(error = %e, "Validacion Firebase ID Token fallo");
349            AppError::Unauthorized
350        })?;
351        let claims = data.claims;
352
353        // 6. Comprobaciones manuales que `jsonwebtoken` no hace:
354        //    - `sub` no vacio (es el firebase_uid).
355        //    - `auth_time` no en el futuro (con tolerancia de reloj).
356        if claims.sub.trim().is_empty() {
357            tracing::warn!("Firebase ID Token con `sub` vacio");
358            return Err(AppError::Unauthorized);
359        }
360        let now = chrono::Utc::now().timestamp();
361        if claims.auth_time > now + CLOCK_SKEW_SECS as i64 {
362            tracing::warn!(auth_time = claims.auth_time, now, "auth_time en el futuro");
363            return Err(AppError::Unauthorized);
364        }
365
366        Ok(claims)
367    }

369    /// Busca una `Jwk` por `kid` en el cache, refrescando si esta caducado.
370    async fn find_jwk(&self, kid: &str) -> Result<Option<Jwk>, AppError> {
371        if self.is_expired().await {
372            self.refresh_jwks().await?;
373        }
374        let cache = self.cache.read().await;
375        Ok(cache.set.as_ref().and_then(|set| set.find(kid).cloned()))
376    }
377
378    /// `true` si el cache esta vacio o ha caducado.
379    async fn is_expired(&self) -> bool {
380        let cache = self.cache.read().await;
381        cache.set.is_none() || Instant::now() >= cache.expires_at
382    }
383
384    /// Refresca el cache de JWKS desde Google.
385    ///
386    /// # Double-checked locking
387    /// Si llegan 50 requests simultaneas con el cache caducado, sin esta
388    /// proteccion las 50 dispararian fetches paralelos al JWKS (thundering
389    /// herd). Con el `refresh_lock`, solo la primera entra; las demas esperan
390    /// el guard, y al entrar comprueban de nuevo `is_expired()`: si el primero
391    /// ya refresco, no vuelven a fetchear.
392    async fn refresh_jwks(&self) -> Result<(), AppError> {
393        let _guard = self.refresh_lock.lock().await;
394
395        // Double-check: el cache puede haberse refrescado mientras esperabamos.
396        if !self.is_expired().await {
397            return Ok(());
398        }
399
400        let resp = self
401            .http
402            .get(self.jwks_url.as_str())
403            .send()
404            .await
405            .map_err(|e| AppError::Internal(format!("error de red al descargar JWKS: {e}")))?
406            .error_for_status()
407            .map_err(|e| AppError::Internal(format!("HTTP no-OK al descargar JWKS: {e}")))?;
408
409        // Respetar `Cache-Control: max-age=N` de Google.
410        // Si la cabecera no se puede parsear, usar FALLBACK_TTL.
411        let ttl = resp
412            .headers()
413            .get(reqwest::header::CACHE_CONTROL)
414            .and_then(|v| v.to_str().ok())
415            .and_then(parse_max_age)
416            .map(Duration::from_secs)
417            .unwrap_or(FALLBACK_TTL);
418
419        let body = resp
420            .bytes()
421            .await
422            .map_err(|e| AppError::Internal(format!("error leyendo cuerpo JWKS: {e}")))?;
423        let set: JwkSet = serde_json::from_slice(&body)
424            .map_err(|e| AppError::Internal(format!("JWKS malformado: {e}")))?;
425
426        let mut cache = self.cache.write().await;
427        cache.set = Some(set);
428        cache.expires_at = Instant::now() + ttl;
429        tracing::info!(ttl_secs = ttl.as_secs(), "JWKS de Firebase refrescado");
430        Ok(())
431    }
432}

434/// Parsea el valor `max-age=N` de un header `Cache-Control`.
435///
436/// Acepta formatos como:
437///   - `public, max-age=21600`
438///   - `max-age=300, must-revalidate`
439///   - `max-age=0` (devuelve 0)
440///
441/// Devuelve `None` si no encuentra `max-age=` o si el numero no es valido.
442fn parse_max_age(header: &str) -> Option<u64> {
443    header
444        .split(',')
445        .map(str::trim)
446        .find_map(|p| p.strip_prefix("max-age=").and_then(|n| n.parse().ok()))
447}
