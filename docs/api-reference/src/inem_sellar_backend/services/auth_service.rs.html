3// Logica de negocio de autenticacion: JWT propios + refresh tokens.
4// La verificacion de Firebase ID Tokens vive en `firebase_verifier.rs`.
5// NO accede a la BD — eso lo hace el repositorio.

8use chrono::{Duration, Utc};
9use jsonwebtoken::{DecodingKey, EncodingKey, Header, Validation, decode, encode};
10use rand::Rng;
11use serde::{Deserialize, Serialize};
12use sha2::{Digest, Sha256};
13use uuid::Uuid;
14
15use crate::errors::AppError;

17/// Claims del JWT — la informacion que va DENTRO del token.
18///
19/// # Por que `sub` y no `id_usuario`
20/// `sub` (subject) es el claim estandar de JWT (RFC 7519) para identificar
21/// al sujeto del token. Usar nombres estandar facilita la interoperabilidad
22/// con librerias y servicios externos.
23///
24/// # Por que `anonimo: bool` con `#[serde(default)]`
25/// Distingue entre usuarios "completos" (Firebase google.com / password)
26/// y usuarios "anonimos" (Firebase Anonymous). El cliente lo usa para
27/// decidir si mostrar el flujo "completar registro".
28///
29/// `#[serde(default)]` garantiza retrocompatibilidad: los tokens emitidos
30/// ANTES de anadir este campo no llevan `anonimo` en el JSON; al
31/// deserializar usaran `false` automaticamente.
32#[derive(Debug, Serialize, Deserialize)]
33pub struct Claims {
34    /// UUID del usuario como string
35    pub sub: String,
36    /// Timestamp de expiracion (segundos desde Unix epoch)
37    pub exp: usize,
38    /// Timestamp de emision
39    pub iat: usize,
40    /// `true` si el token corresponde a un usuario anonimo.
41    /// El default `false` se aplica si el campo no esta presente (tokens
42    /// emitidos antes de la integracion con Firebase).
43    #[serde(default)]
44    pub anonimo: bool,
45}

47/// Servicio de autenticacion — encapsula toda la criptografia y logica de tokens.
48///
49/// # Por que un struct y no funciones sueltas
50/// El servicio necesita configuracion (jwt_secret, expiracion). En vez de pasar
51/// estos valores como parametros en cada llamada, los almacenamos en el struct.
52/// Es el patron "Service Object" — equivalente a un servicio inyectado con
53/// Provider/GetIt en Flutter.
54///
55/// # Por que `Clone`
56/// Salvo necesita que los tipos inyectados en Depot sean `Clone`.
57/// `String` es Clone, `u64` es Copy (aun mas barato que Clone).
58#[derive(Clone)]
59pub struct AuthService {
60    jwt_secret: String,
61    jwt_expiracion_minutos: u64,
62}
63
64impl AuthService {
65    /// Crea un nuevo servicio con la configuracion JWT dada.
66    pub fn new(jwt_secret: String, jwt_expiracion_minutos: u64) -> Self {
67        Self {
68            jwt_secret,
69            jwt_expiracion_minutos,
70        }
71    }
72
73    /// Genera un access token JWT firmado con HS256 marcando si el usuario
74    /// es anonimo.
75    ///
76    /// Usado por:
77    ///   - `/auth/firebase` con el flag derivado del `sign_in_provider`
78    ///     (`true` solo para Firebase Anonymous).
79    ///   - `/auth/refrescar` preservando el flag del usuario consultando
80    ///     `proveedor_repo.es_anonimo()`.
81    ///
82    /// # Por que `anonimo` va en el JWT y no se consulta la BD en cada request
83    /// El JWT se valida sin tocar BD (criptografia pura). Si tuvieramos que
84    /// consultar la BD para saber si el usuario es anonimo, anadiriamos una
85    /// query a CADA request protegida — innecesario porque el flag no cambia
86    /// hasta que el usuario "se complete" (futuro endpoint `/auth/upgrade`),
87    /// momento en el que se emite un nuevo JWT con `anonimo=false`.
88    pub fn generar_access_token_con_flag(
89        &self,
90        id_usuario: Uuid,
91        anonimo: bool,
92    ) -> Result<String, AppError> {
93        let ahora = Utc::now();
94        let expiracion = ahora + Duration::minutes(self.jwt_expiracion_minutos as i64);
95
96        let claims = Claims {
97            sub: id_usuario.to_string(),
98            exp: expiracion.timestamp() as usize,
99            iat: ahora.timestamp() as usize,
100            anonimo,
101        };
102
103        encode(
104            &Header::default(),
105            &claims,
106            &EncodingKey::from_secret(self.jwt_secret.as_bytes()),
107        )
108        .map_err(|e| AppError::Internal(format!("Error generando JWT: {e}")))
109    }
110
111    /// Verifica y decodifica un access token JWT.
112    /// Devuelve los Claims si el token es valido y no ha expirado.
113    pub fn verificar_access_token(&self, token: &str) -> Result<Claims, AppError> {
114        let token_data = decode::<Claims>(
115            token,
116            &DecodingKey::from_secret(self.jwt_secret.as_bytes()),
117            &Validation::default(),
118        )
119        .map_err(|_| AppError::Unauthorized)?;
120
121        Ok(token_data.claims)
122    }
123
124    /// Genera un refresh token aleatorio (32 bytes, codificado en hex = 64 chars).
125    ///
126    /// # Por que hex y no base64
127    /// Hex es mas simple, no tiene caracteres especiales (+, /, =) que
128    /// podrian causar problemas en URLs o JSON. 64 chars hex = 256 bits
129    /// de entropia, mas que suficiente.
130    pub fn generar_refresh_token(&self) -> String {
131        let bytes: [u8; 32] = rand::thread_rng().r#gen();
132        bytes.iter().map(|b| format!("{:02x}", b)).collect()
133    }
134
135    /// Hashea un refresh token con SHA-256 para almacenarlo en la BD.
136    ///
137    /// # Por que SHA-256 y no Argon2 para refresh tokens
138    /// Los refresh tokens ya tienen alta entropia (256 bits aleatorios).
139    /// No necesitan un hash lento (Argon2) porque un atacante no puede
140    /// hacer ataques de diccionario contra bytes aleatorios. SHA-256 es
141    /// suficiente y mucho mas rapido.
142    pub fn hashear_refresh_token(&self, token: &str) -> String {
143        let mut hasher = Sha256::new();
144        hasher.update(token.as_bytes());
145        format!("{:x}", hasher.finalize())
146    }
147}
