6use serde::Deserialize;
7
8/// Configuracion global de la aplicacion, cargada desde variables de entorno.
9///
10/// # Por que un struct con Deserialize
11/// En Rust, `struct` agrupa datos relacionados (similar a una clase en Dart pero sin herencia).
12/// `#[derive(Deserialize)]` genera automaticamente el codigo para convertir datos externos
13/// (variables de entorno, JSON) en este struct — como un `fromJson()` en Dart, pero resuelto
14/// en tiempo de compilacion (coste cero en runtime).
15///
16/// # Por que Clone
17/// `Clone` permite crear copias independientes del struct. Lo necesitamos porque la config
18/// se comparte entre multiples handlers async, y cada uno necesita su propia copia.
19/// En Dart todos los objetos viven en el heap y el garbage collector gestiona la memoria;
20/// en Rust, nosotros decidimos explicitamente cuando copiar datos.
21#[derive(Deserialize, Clone)]
22pub struct AppConfig {
23    /// URL de conexion a PostgreSQL. Ejemplo: `postgres://user:pass@host/db`.
24    /// Se lee de la variable de entorno `DATABASE_URL`.
25    pub database_url: String,
26
27    /// Direccion IP y puerto donde escucha el servidor.
28    /// Valor por defecto: `0.0.0.0:8080` (todas las interfaces, puerto 8080).
29    ///
30    /// # Por que String y no &str
31    /// `String` es un tipo "owned" — el struct es dueno del dato y lo mantiene en memoria.
32    /// `&str` es una referencia prestada (borrowed) que necesitaria un lifetime `'a`,
33    /// complicando el codigo. Regla practica: si el dato debe vivir tanto como el struct, usa `String`.
34    #[serde(default = "default_addr")]
35    pub server_addr: String,
36
37    /// Direccion de binding completa del servidor en formato `IP:puerto`.
38    /// Valor por defecto: `0.0.0.0:8080`.
39    ///
40    /// Se lee de la variable de entorno `PORT_ADDR`. Se usa cuando se necesita
41    /// distinguir entre la direccion de escucha del socket TCP (`port_addr`,
42    /// que usa `0.0.0.0` para aceptar todas las interfaces) y la direccion
43    /// publica que se incluye en los logs (`server_addr`, que muestra `localhost`
44    /// para que sea clickable en el terminal durante desarrollo).
45    #[serde(default = "port_addr")]
46    pub port_addr: String,
47
48    /// Secreto para firmar/verificar tokens JWT (HS256).
49    /// Se lee de la variable de entorno `JWT_SECRET`.
50    /// CRITICO: en produccion debe ser un valor aleatorio largo (>= 32 chars).
51    pub jwt_secret: String,
52
53    /// Duracion del access token JWT en minutos.
54    /// Default: 15 minutos. El frontend debe refrescar antes de que expire.
55    #[serde(default = "default_jwt_expiracion")]
56    pub jwt_expiracion_minutos: u64,
57
58    /// Identificador del proyecto de Firebase (`projectId` de la consola).
59    ///
60    /// # Por que es obligatorio y no tiene default
61    /// El verificador de Firebase ID Tokens valida que el claim `aud` (audience)
62    /// del token coincida EXACTAMENTE con este valor, y que el claim `iss` sea
63    /// `https://securetoken.google.com/<project_id>`. Si arrancamos el servidor
64    /// con un valor incorrecto o vacio, todos los logins con Firebase fallarian
65    /// silenciosamente con 401. Es preferible un panic claro al arranque.
66    ///
67    /// # De donde sale
68    /// Es el mismo `project_id` que figura en `google-services.json` (Android)
69    /// y `GoogleService-Info.plist` (iOS) del cliente Flutter.
70    /// Se lee de la variable de entorno `FIREBASE_PROJECT_ID`.
71    pub firebase_project_id: String,
72
73    // ── Configuracion SMTP (notificaciones de reportes al admin) ────────────
74    //
75    // Todos los campos son opcionales (default: cadena vacia / 587). Si
76    // `smtp_password` o `report_email_to` quedan vacios, el `EmailNotifier`
77    // se construye en modo deshabilitado y los handlers siguen funcionando
78    // sin enviar email — util para arrancar el servicio en dev sin SMTP.
79    /// Host del servidor SMTP. Se lee de `SMTP_HOST` (ej. `smtp.gmail.com`).
80    #[serde(default)]
81    pub smtp_host: String,
82
83    /// Puerto del servidor SMTP. Default: `587` (STARTTLS).
84    /// Se lee de `SMTP_PORT`.
85    #[serde(default = "default_smtp_port")]
86    pub smtp_port: u16,
87
88    /// Usuario para autenticarse en el servidor SMTP.
89    /// En Gmail Workspace coincide con la direccion de correo del remitente.
90    /// Se lee de `SMTP_USER`.
91    #[serde(default)]
92    pub smtp_user: String,
93
94    /// Contrasena (o app password en Gmail) del usuario SMTP.
95    /// CRITICO: en produccion debe inyectarse via `Secret` de Kubernetes,
96    /// nunca commitearse al repositorio. Se lee de `SMTP_PASSWORD`.
97    #[serde(default)]
98    pub smtp_password: String,
99
100    /// Direccion `From:` que aparecera en el correo al admin.
101    /// Debe coincidir con `smtp_user` (o ser un alias send-as configurado)
102    /// para que Gmail no rechace el envio. Se lee de `REPORT_EMAIL_FROM`.
103    #[serde(default)]
104    pub report_email_from: String,
105
106    /// Direccion `To:` del admin que recibe la notificacion.
107    /// Si esta vacio, las notificaciones de reportes quedan deshabilitadas.
108    /// Se lee de `REPORT_EMAIL_TO`.
109    #[serde(default)]
110    pub report_email_to: String,
111}
112
113/// Puerto SMTP por defecto: `587` (submission con STARTTLS).
114///
115/// # Por que 587 y no 465 ni 25
116/// `587` es el puerto de "submission" (RFC 6409): autenticado y con STARTTLS.
117/// `465` es SMTPS (TLS implicito) — Gmail tambien lo soporta, pero `587` es
118/// mas estandar. `25` es relay entre servidores y casi todos los proveedores
119/// lo bloquean para clientes (anti-spam).
120fn default_smtp_port() -> u16 {
121    587
122}
123
124/// Duracion por defecto del access token: 15 minutos.
125fn default_jwt_expiracion() -> u64 {
126    15
127}
128
129/// Devuelve la direccion por defecto del servidor.
130///
131/// # Por que una funcion separada
132/// `serde` requiere que el valor por defecto de un campo sea una funcion (no un literal).
133/// `#[serde(default = "default_addr")]` llama a esta funcion si la variable de entorno
134/// `SERVER_ADDR` no esta definida. Es una limitacion del sistema de macros de serde.
135///
136/// # Por que `.to_string()`
137/// `"0.0.0.0:8080"` es un `&str` (referencia a texto en el binario), pero necesitamos
138/// un `String` (texto en el heap, owned). `.to_string()` crea esa copia en el heap.
139fn default_addr() -> String {
140    "localhost:8080".to_string()
141}
142
143/// Devuelve la direccion de binding por defecto del servidor.
144///
145/// Misma logica que `default_addr`: serde necesita una funcion, no un literal.
146/// `0.0.0.0` significa "escucha en todas las interfaces de red disponibles",
147/// lo que es necesario dentro de un contenedor Docker para que el trafico
148/// externo pueda llegar al proceso.
149fn port_addr() -> String {
150    "0.0.0.0:8080".to_string()
151}
152
153impl AppConfig {
154    /// Carga la configuracion desde variables de entorno.
155    ///
156    /// Primero carga el archivo `.env` si existe (via `dotenvy`), luego deserializa
157    /// todas las variables de entorno en un `AppConfig` (via `envy`).
158    ///
159    /// # Por que `dotenvy::dotenv().ok()`
160    /// `.ok()` convierte un `Result` en `Option`, descartando el error silenciosamente.
161    /// Esto es intencionado: en produccion no habra archivo `.env` (las variables vienen
162    /// del entorno del contenedor Docker), asi que no queremos que falle.
163    ///
164    /// # Por que `.expect()` y no `?`
165    /// `expect()` hace panic con un mensaje claro si falla. Aqui es aceptable porque
166    /// la configuracion es critica: sin ella, el servidor no puede arrancar. Es mejor
167    /// fallar rapido con un mensaje claro que intentar continuar sin config.
168    /// En handlers de API, usaremos `?` para propagar errores sin panic.
169    pub fn from_env() -> Self {
170        dotenvy::dotenv().ok();
171        envy::from_env::<Self>().expect("Error cargando configuracion desde variables de entorno")
172    }
173}
